186 , start_time(0.0)

188 , counter(0)
189 , module(m)
190 , timer_status(UWOFDMALOHA_IDLE)
191 {
192 assert(m != NULL);

198 virtual ~UWOFDMAlohaTimer()
199 {
200 }

209 virtual void
210 freeze()
211 {
212 assert(timer_status == UWOFDMALOHA_RUNNING);
213 left_duration -= (NOW - start_time);
214 if (left_duration <= 0.0)
215 left_duration = module->mac2phy_delay_;
216 force_cancel();
217 timer_status = UWOFDMALOHA_FROZEN;
218 }
223 virtual void
224 unFreeze()
225 {
226 assert(timer_status == UWOFDMALOHA_FROZEN);
227 start_time = NOW;
228 assert(left_duration > 0);
229 sched(left_duration);
230 timer_status = UWOFDMALOHA_RUNNING;
231 }
232
236 virtual void
237 stop()
238 {
239 timer_status = UWOFDMALOHA_IDLE;
240 force_cancel();
241 }
242
247 virtual void
248 schedule(double val)
249 {
250 start_time = NOW;
251 left_duration = val;
252 timer_status = UWOFDMALOHA_RUNNING;
253 resched(val);
254 }
255
260 bool
261 isIdle()
262 {
263 return (timer_status == UWOFDMALOHA_IDLE);
264 }

271 isRunning()

273 return (timer_status == UWOFDMALOHA_RUNNING);

301 isActive()

304 timer_status == UWOFDMALOHA_RUNNING);
305 }
