216 virtual void
217 freeze()
218 {
219 assert(timer_status == UWOFDMALOHA_RUNNING);
220 left_duration -= (NOW - start_time);
221 if (left_duration <= 0.0)
222 left_duration = module->mac2phy_delay_;
223 force_cancel();
224 timer_status = UWOFDMALOHA_FROZEN;
225 }
230 virtual void
231 unFreeze()
232 {
233 assert(timer_status == UWOFDMALOHA_FROZEN);
234 start_time = NOW;
235 assert(left_duration > 0);
236 sched(left_duration);
237 timer_status = UWOFDMALOHA_RUNNING;
238 }
239
243 virtual void
244 stop()
245 {
246 timer_status = UWOFDMALOHA_IDLE;
247 force_cancel();
248 }
249
254 virtual void
255 schedule(double val)
256 {
257 start_time = NOW;
258 left_duration = val;
259 timer_status = UWOFDMALOHA_RUNNING;
260 resched(val);
261 }

268 isIdle()
