152  virtual void
153  freeze()
154  {
155  assert(timer_status == UW_CS_ALOHA_TRIG_SINK_RUNNING);
156  left_duration -= (NOW - start_time);
157  if (left_duration <= 0.0)
158  left_duration = module->mac2phy_delay_;
159  force_cancel();
160  timer_status = UW_CS_ALOHA_TRIG_SINK_FROZEN;
161  }
162 
167  virtual void
168  unFreeze()
169  {
170  assert(timer_status == UW_CS_ALOHA_TRIG_SINK_FROZEN);
171  start_time = NOW;
172  assert(left_duration > 0);
173  sched(left_duration);
174  timer_status = UW_CS_ALOHA_TRIG_SINK_RUNNING;
175  }
176 
180  virtual void
181  stop()

183  timer_status = UW_CS_ALOHA_TRIG_SINK_IDLE;
184  force_cancel();
185  }
