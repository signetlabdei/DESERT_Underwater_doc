269 void
270 uwApplicationModule::statistics(Packet *p)
271 {
272  hdr_cmn *ch = hdr_cmn::access(p);
273  hdr_DATA_APPLICATION *uwApph = HDR_DATA_APPLICATION(p);
274 
275  if (ch->ptype_ != PT_DATA_APPLICATION) {
276  if (debug_ >= 0)
277  std::cout << "[" << getEpoch() << "]::" << NOW
278  << "::UWAPPLICATION::DROP_PACKET_NOT_APPLICATION_TYPE"
279  << endl;
280  if (logging)
281  out_log << left << "[" << getEpoch() << "]::" << NOW
282  << "::UWAPPLICATION::DROP_PACKET_NOT_APPLICATION_TYPE"
283  << endl;
284  drop(p, 1, UWAPPLICATION_DROP_REASON_UNKNOWN_TYPE);
285  incrPktInvalid(); // Increment the number of packet received invalid
286  return;
287  }
288  esn = hrsn + 1; // Increase the expected sequence number
289 
290  // Verify if the data packet is already processed.
291  if (useDropOutOfOrder()) {
292  if (sn_check[uwApph->sn_ &
293  0x00ffffff]) { // Packet already processed: drop it
294  if (debug_ >= 0)
295  std::cout << "[" << getEpoch() << "]::" << NOW
296  << "::UWAPPLICATION::DROP_PACKET_PACKET_ALREADY_"
297  "PROCESSED_ID_"
298  << (int) uwApph->sn_ << endl;
299  if (logging)
300  out_log << left << "[" << getEpoch() << "]::" << NOW
301  << "::UWAPPLICATION::DROP_PACKET_PACKET_ALREADY_"
302  "PROCESSED_ID_"
303  << (int) uwApph->sn_ << endl;
304  incrPktInvalid();
305  drop(p, 1, UWAPPLICATION_DROP_REASON_DUPLICATED_PACKET);
306  return;
307  }
308  }
309  // The data packet with this particular SN is not already processed
310  // Set true sn_check. In this way we assume that these sn are already
311  // processed by the node
312  sn_check[uwApph->sn_ & 0x00ffffff] = true;
313 
314  // The data packet received is out of order
315  if (useDropOutOfOrder()) {
316  if (uwApph->sn_ <
317  esn) { // packet is out of sequence and is to be discarded
318  incrPktOoseq(); // Increase the number of data packets receive out
319  // of
320  // sequence.
321  if (debug_ >= 0)
322  std::cout << "[" << getEpoch() << "]::" << NOW
323  << "::UWAPPLICATION::DROP_PACKET_PACKET_OOS_ID_"
324  << (int) uwApph->sn_ << "_LAST_SN_" << hrsn << endl;
325  if (logging)
326  out_log << left << "[" << getEpoch() << "]::" << NOW
327  << "::UWAPPLICATION::DROP_PACKET_PACKET_OOS_ID_"
328  << (int) uwApph->sn_ << "_LAST_SN_" << hrsn << endl;
329  drop(p, 1, UWAPPLICATION_DROP_REASON_OUT_OF_SEQUENCE);
330  return;
331  }
332  }
333 
334  // Compute the Forward Trip time
335  rftt = Scheduler::instance().clock() - ch->timestamp();
336  if ((uwApph->rftt_valid_) / 10000) {
337  double rtt = rftt + uwApph->rftt();
338  updateRTT(rtt);
339  }
340 
341  updateFTT(rftt); // Update the forward trip time
342 
343  hrsn = uwApph->sn_; // Update the highest sequence number received
344 
345  // Verify if a packet is lost
346  if (useDropOutOfOrder()) {
347  if (uwApph->sn_ > esn) {
348  if (debug_ >= 0)
349  std::cout << "[" << getEpoch() << "]::" << NOW
350  << "::UWAPPLICATION::PACKET_LOST_ID_RECEIVED"
351  << (int) uwApph->sn_ << "_ID_EXPECTED_" << esn
352  << endl;
353  if (logging)
354  out_log << left << "[" << getEpoch() << "]::" << NOW
355  << "::UWAPPLICATION::PACKET_LOST_ID_RECEIVED"
356  << (int) uwApph->sn_ << "_ID_EXPECTED_" << esn << endl;
357  incrPktLost(uwApph->sn_ - (esn));
358  }
359  }
360 
361  double dt = Scheduler::instance().clock() - lrtime;
362  // updateThroughput(ch->size(), dt); //Update Throughput
363  updateThroughput(uwApph->payload_size(), dt);
364  incrPktRecv(); // Increase the number of data packets received
365 
366  lrtime = Scheduler::instance().clock(); // Update the time in which the last
367  // packet is received.
368  if (debug_ >= 0 && socket_active) {
369  std::cout << "[" << getEpoch() << "]::" << NOW
370  << "::UWAPPLICATION::PAYLOAD_RECEIVED--> ";
371  for (int i = 0; i < uwApph->payload_size(); i++) {
372  cout << uwApph->payload_msg[i];
373  }
374  }
375  if (debug_ >= 0)
376  std::cout << "[" << getEpoch() << "]::" << NOW
377  << "::UWAPPLICATION::SN_RECEIVED_" << (int) uwApph->sn_
378  << endl;
379  if (debug_ >= 0)
380  std::cout << "[" << getEpoch() << "]::" << NOW
381  << "::UWAPPLICATION::PAYLOAD_SIZE_RECEIVED_"
382  << (int) uwApph->payload_size() << endl;
383  if (debug_ >= 1 && !withoutSocket())
384  std::cout << "[" << getEpoch() << "]::" << NOW
385  << "::UWAPPLICATION::PAYLOAD_RECEIVED_" << uwApph->payload_msg
386  << endl;
387 
388  if (logging)
389  out_log << left << "[" << getEpoch() << "]::" << NOW
390  << "::UWAPPLICATION::PAYLOAD_SIZE_RECEIVED_"
391  << (int) uwApph->payload_size() << endl;
392  if (logging)
393  out_log << left << "[" << getEpoch() << "]::" << NOW
394  << "::UWAPPLICATION::SN_RECEIVED_" << (int) uwApph->sn_ << endl;
395 
396  if (logging && !withoutSocket()) {
397  out_log << left << "::" << NOW
398  << "::UWAPPLICATION::PAYLOAD_RECEIVED--> ";
399  for (int i = 0; i < uwApph->payload_size(); i++) {
400  out_log << uwApph->payload_msg[i];
401  }
402  out_log << std::endl;
403  }
404  if (clnSockDescr) {
405  write(clnSockDescr, uwApph->payload_msg, (size_t)uwApph->payload_size());
406  }
407  Packet::free(p);
408 } // end statistics method

422 void
423 uwApplicationModule::init_Packet()
424 {
425  Packet *p = Packet::alloc();
426 
427  double delay = 0;
428 
429  hdr_cmn *ch = hdr_cmn::access(p);
430  hdr_uwudp *uwudp = hdr_uwudp::access(p);
431  hdr_uwip *uwiph = hdr_uwip::access(p);
432  hdr_DATA_APPLICATION *uwApph = HDR_DATA_APPLICATION(p);
433 
434  // Common header fields
435  ch->uid() = uidcnt++; // Increase the id of data packet
436  ch->ptype_ = PT_DATA_APPLICATION; // Assign the type of packet that is being
437  // created
438  ch->size() = payloadsize; // Assign the size of data payload
439  uwApph->payload_size() = payloadsize;
440  ch->direction() = hdr_cmn::DOWN; // The packet must be forward at the level
441  // above of him
442 
443  // Transport header fields
444  uwudp->dport() = port_num; // Set the destination port
445 
446  // IP header fields
447  uwiph->daddr() = dst_addr; // Set the IP destination address
448 
449  // Application header fields
450  incrPktSent();
451  uwApph->sn_ = txsn; // Sequence number to the data packet
452 
453  if (rftt >= 0) {
454  uwApph->rftt_ = (int) (rftt * 10000); // Forward Trip Time
455  uwApph->rftt_valid_ = true;
456  } else {
457  uwApph->rftt_valid_ = false;
458  }
459  uwApph->priority_ = 0; // Priority of the message
460 
461  // Create the payload message
462  if (getpayloadsize() < MAX_LENGTH_PAYLOAD) {
463  for (int i = 0; i < getpayloadsize(); i++) {
464  (*uwApph).payload_msg[i] = rand() % 26 + 'a';
465  }
466  for (int i = getpayloadsize(); i < MAX_LENGTH_PAYLOAD; i++) {
467  (*uwApph).payload_msg[i] = '0';
468  }
469  } else {
470  for (int i = 0; i < MAX_LENGTH_PAYLOAD; i++) {
471  (*uwApph).payload_msg[i] = rand() % 26 + 'a';
472  }
473  }
474 
475  // Show the DATA payload generated
476  ch->timestamp() = Scheduler::instance().clock();
477 
478  // Show some DATA packet information
479  if (debug_ >= 2)
480  std::cout << "[" << getEpoch() << "]::" << NOW
481  << "::UWAPPLICATION::INIT_PACKET::UID_" << ch->uid_ << endl;
482  if (debug_ >= 0)
483  std::cout << "[" << getEpoch() << "]::" << NOW
484  << "::UWAPPLICATION::INIT_PACKET::DEST_"
485  << (int) uwiph->daddr() << endl;
486  if (debug_ >= 0)
487  std::cout << "[" << getEpoch() << "]::" << NOW
488  << "::UWAPPLICATION::INIT_PACKET::SIZE_"
489  << (int) uwApph->payload_size() << endl;
490  if (debug_ >= 0)
491  std::cout << "[" << getEpoch() << "]::" << NOW
492  << "::UWAPPLICATION::INIT_PACKET::SN_" << (int) uwApph->sn_
493  << endl;
494  if (debug_ >= 0)
495  std::cout << "[" << getEpoch() << "]::" << NOW
496  << "::UWAPPLICATION::INIT_PACKET::SEND_DOWN_PACKET" << endl;
497 
498  if (logging)
499  out_log << left << "[" << getEpoch() << "]::" << NOW
500  << "::UWAPPLICATION::INIT_PACKET::UID_" << ch->uid_ << endl;
501  if (logging)
502  out_log << left << "[" << getEpoch() << "]::" << NOW
503  << "::UWAPPLICATION::INIT_PACKET::DEST_" << (int) uwiph->daddr()
504  << endl;
505  if (logging)
506  out_log << left << "[" << getEpoch() << "]::" << NOW
507  << "::UWAPPLICATION::INIT_PACKET::SIZE_"
508  << (int) uwApph->payload_size() << endl;
509  if (logging)
510  out_log << left << "[" << getEpoch() << "]::" << NOW
511  << "::UWAPPLICATION::INIT_PACKET::SN_" << (int) uwApph->sn_
512  << endl;
513  if (logging)
514  out_log << left << "[" << getEpoch() << "]::" << NOW
515  << "::UWAPPLICATION::INIT_PACKET::SEND_DOWN_PACKET" << endl;
516  sendDown(p, delay);
517  chkTimerPeriod.resched(
518  getTimeBeforeNextPkt()); // schedule next transmission
519 } // end init_Packet() method

535 double
536 uwApplicationModule::getTimeBeforeNextPkt()
537 {
538  if (getPeriod() < 0) {
539  if (debug_ >= 0)
540  std::cout << "[" << getEpoch() << "]::" << NOW
541  << "::UWAPPLICATION::ERROR_TIMER SET TO NEGATIVE VALUE"
542  << endl;
543  exit(1);
544  }
545  if (usePoissonTraffic()) {
546  // Data packets are generated with a Poisson process
547  double u = RNG::defaultrng()->uniform_double();
548  double lambda = 1.0 / PERIOD;
549  if (debug_ >= 2)
550  std::cout
551  << "[" << getEpoch() << "]::" << NOW
552  << "::UWAPPLICATION::PACKET_GENERATED_WITH_POISSON_PERIOD_"
553  << PERIOD << endl;
554  if (logging)
555  out_log << left << "[" << getEpoch() << "]::" << NOW
556  << "::UWAPPLICATION::PACKET_GENERATED_WITH_POISSON_PERIOD_"
557  << PERIOD << endl;
558  return (-log(u) / lambda);
559  } else {
560  // Data packets are generated wit a costant bit rate
561  if (debug_ >= 2)
562  std::cout << "[" << getEpoch() << "]::" << NOW
563  << "::UWAPPLICATION::PACKET_GENERATED_WITH_FIXED_PERIOD_"
564  << PERIOD << endl;
565  if (logging)
566  out_log << left << "[" << getEpoch() << "]::" << NOW
567  << "::UWAPPLICATION::PACKET_GENERATED_WITH_FIXED_PERIOD_"
568  << PERIOD << endl;
569  return PERIOD;
570  }
571 } // end getTimeBeforeNextPkt() Method

579 double
580 uwApplicationModule::GetRTTstd() const
581 {

583  double var =
584  (sumrtt2 - (sumrtt * sumrtt / rttsamples)) / (rttsamples - 1);

586  } else
587  return 0;

608  double var = 0;

612  else
613  return 0;
614  } else {
615  return 0;
