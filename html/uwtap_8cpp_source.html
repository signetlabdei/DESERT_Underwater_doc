65int
66UwTap::recvSyncClMsg(ClMessage *m)
67{
68 // TAP is supposed to be transparent to other layers
69 if (m->direction() == DOWN) {
70 sendSyncClMsgDown(m);
71 return 0;
72 } else if (m->direction() == UP) {
73 sendSyncClMsgUp(m);
74 return 0;
75 }
76
77 return 0;
78}

80void
81UwTap::recv(Packet *p)
82{
83 hdr_cmn *ch = HDR_CMN(p);
84 if (ch->direction() == hdr_cmn::UP) {
85 // ch->timestamp() = NOW;
86 ClMsgTapPkt msg = ClMsgTapPkt(
87 p, clmsg_tap_type::END_RX, NOW, Tap2PhyTxDuration(p));
88 sendSyncClMsg(&msg);
89 sendUp(p);
90 } else if (ch->direction() == hdr_cmn::DOWN) {
91 // ch->txtime() = Tap2PhyTxDuration(p);
92 ClMsgTapPkt msg = ClMsgTapPkt(
93 p, clmsg_tap_type::START_TX, NOW, Tap2PhyTxDuration(p));
94 sendSyncClMsg(&msg);
95 sendDown(p);
96 }
97}
