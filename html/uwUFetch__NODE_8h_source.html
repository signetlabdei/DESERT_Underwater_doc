264  : TimerHandler()
265  , start_time(0.0)

282  virtual void
283  freeze()
284  {
285  assert(timer_status == UWUFETCH_TIMER_STATUS_RUNNING);
286  left_duration = NOW - start_time;
287  if (left_duration <= 0.0) {
288  left_duration = module->mac2phy_delay_;
289  }
290  force_cancel();
291  timer_status = UWUFETCH_TIMER_STATUS_FROZEN;
292  }
293 
298  virtual void
299  unFreeze()
300  {
301  assert(timer_status == UWUFETCH_TIMER_STATUS_FROZEN);
302  start_time = NOW;
303  assert(left_duration > 0);
304  sched(left_duration);
308  timer_status = UWUFETCH_TIMER_STATUS_RUNNING;
309  }
310 
314  virtual void
315  stop()
316  {
317  timer_status = UWUFETCH_TIMER_STATUS_IDLE;
318  force_cancel();
319  }
320 
326  virtual void
327  schedule(double val)
328  {
329  start_time = NOW;
330  left_duration = val;
331  timer_status = UWUFETCH_TIMER_STATUS_RUNNING;
332  resched(val);
333  }
