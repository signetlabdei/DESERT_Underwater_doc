256  class uwUFetch_NODE_timer : public TimerHandler
257  {
258  public:
263  uwUFetch_NODE_timer(uwUFetch_NODE *m)
264  : TimerHandler()
265  , start_time(0.0)
266  , left_duration(0.0)
267  , counter(0)
268  , module(m)
269  , timer_status(UWUFETCH_TIMER_STATUS_IDLE)
270  {
271  assert(m != NULL);
272  }
273 
277  virtual ~uwUFetch_NODE_timer(){};
278 
282  virtual void
283  freeze()
284  {
285  assert(timer_status == UWUFETCH_TIMER_STATUS_RUNNING);
286  left_duration = NOW - start_time;
287  if (left_duration <= 0.0) {
288  left_duration = module->mac2phy_delay_;
289  }
290  force_cancel();
291  timer_status = UWUFETCH_TIMER_STATUS_FROZEN;
292  }
293 
298  virtual void
299  unFreeze()
300  {
301  assert(timer_status == UWUFETCH_TIMER_STATUS_FROZEN);
302  start_time = NOW;
303  assert(left_duration > 0);
304  sched(left_duration);
308  timer_status = UWUFETCH_TIMER_STATUS_RUNNING;
309  }
310 
314  virtual void
315  stop()
316  {
317  timer_status = UWUFETCH_TIMER_STATUS_IDLE;
318  force_cancel();
319  }
320 
326  virtual void
327  schedule(double val)
328  {
329  start_time = NOW;
330  left_duration = val;
331  timer_status = UWUFETCH_TIMER_STATUS_RUNNING;
332  resched(val);
333  }
334 
340  bool
341  isIdle()
342  {
343  return (timer_status == UWUFETCH_TIMER_STATUS_IDLE);
344  }
345 
351  bool
352  isRunning()
353  {
354  return (timer_status == UWUFETCH_TIMER_STATUS_RUNNING);
355  }
356 
362  bool
363  isFroozen()
364  {
365  return (timer_status == UWUFETCH_TIMER_STATUS_FROZEN);
366  }
367 
373  bool
374  isExpired()
375  {
376  return (timer_status == UWUFETCH_TIMER_STATUS_EXPIRED);
377  }
378 
384  bool
385  isActive()
386  {
387  return (timer_status == isRunning() || timer_status == isFroozen());
388  }
389 
393  void
394  resetCounter()
395  {
396  counter = 0;
397  }
398 
402  void
403  incrCounter()
404  {
405  ++counter;
406  }
407 
413  int
414  getCounter()
415  {
416  return counter;
417  }
418 
424  double
425  getDuration()
426  {
427  return left_duration;
428  }
429 
430  protected:
431  double start_time;
432  double left_duration;
433  int counter;

438  }; // END class uwUFetch_NODE_timer
