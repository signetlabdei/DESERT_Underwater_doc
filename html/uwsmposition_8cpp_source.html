173 void
174 UWSMPosition::update(double now)
175 {
176  if ((trgTime_ < 0.) || (now < lastUpdateTime_ + 1e-6))
177  return;
178  double gamma;
179  double theta;
180  double theta_den = sqrt(pow(Ydest_ - Ysorg_, 2.0) +
181  pow(Xdest_ - Xsorg_, 2.0) + pow(Zdest_ - Zsorg_, 2.0));
182  if (theta_den == 0) {
183  x_ = Xsorg_;
184  y_ = Ysorg_;
185  z_ = Zsorg_;
186  } else {
187  theta = acos((Zdest_ - Zsorg_) / theta_den);
188  if (Xdest_ - Xsorg_ == 0)
189  gamma = pi / 2 * sgn(Ydest_ - Ysorg_);
190  else
191  gamma = atan((Ydest_ - Ysorg_) / (Xdest_ - Xsorg_));
192  if ((Xdest_ - Xsorg_) < 0.0)
193  gamma += (Ysorg_ - Ydest_) >= 0.0 ? pi : -pi;
194  x_ = Xsorg_ + (speed_ * (now - trgTime_)) * sin(theta) * cos(gamma);
195  y_ = Ysorg_ + (speed_ * (now - trgTime_)) * sin(theta) * sin(gamma);
196  z_ = Zsorg_ + (speed_ * (now - trgTime_)) * cos(theta);
197  if (pow(Ydest_ - Ysorg_, 2.0) + pow(Xdest_ - Xsorg_, 2.0) +
198  pow(Zdest_ - Zsorg_, 2.0) <
199  pow(x_ - Xsorg_, 2.0) + pow(y_ - Ysorg_, 2.0) +
200  pow(z_ - Zsorg_, 2.0)) {
201  x_ = Xdest_;
202  y_ = Ydest_;
203  z_ = Zdest_;
204  }
205  if (debug_)
206  printf("New pos (%f,%f,%f), dest(%f,%f,%f), source(%f,%f,%f), speed %f sen(%f)=%f\n",
207  x_,
208  y_,
209  z_,
210  Xdest_,
211  Ydest_,
212  Zdest_,
213  Xsorg_,
214  Ysorg_,
215  Zsorg_,
216  speed_,
217  gamma,
218  sin(gamma));
219  }
220 
221  lastUpdateTime_ = now;
222 }
