65UwElectroMagneticChannel::UwElectroMagneticChannel()
66 : ChannelModule()
67 , refractive_index(REFRACTIVE_INDEX_WATER)
68 , speed_of_light(SPEED_OF_LIGHT_VACUUM)
69{
70 bind("RefractiveIndex_", (double *) &refractive_index);
71
72 if (refractive_index < REFRACTIVE_INDEX_MIN) {
73 refractive_index = REFRACTIVE_INDEX_MIN;
74 }
75
76 speed_of_light /= refractive_index;

86void
87UwElectroMagneticChannel::sendUpPhy(Packet *p, ChSAP *chsap)
88{
89 Scheduler &s = Scheduler::instance();
90 struct hdr_cmn *hdr = HDR_CMN(p);
91
92 hdr->direction() = hdr_cmn::UP;
93
94 Position *sourcePos = chsap->getPosition();
95 ChSAP *dest;
96
97 if (debug_)
98 cout << "UwElectroMagneticChannel::sendUpPhy() sending packet" << endl;
99
100 for (int i = 0; i < getChSAPnum(); i++) {
101 dest = (ChSAP *) getChSAP(i);
102
103 if (chsap == dest) // it's the source node -> skip it
104 continue;
105
106 s.schedule(
107 dest, p->copy(), getPropDelay(sourcePos, dest->getPosition()));
108 }
109
110 Packet::free(p);
111}

119double
120UwElectroMagneticChannel::getPropDelay(Position *src, Position *dst)
121{
122 double distance = src->getDist(dst);
123
124 assert(distance >= 0.0);
125
126 double delay = distance / speed_of_light;
127
128 if (debug_)
129 cout << "UwElectroMagneticChannel::getPropDelay() distance = " << distance
130 << "; speed = " << speed_of_light << "; delay = " << delay << endl;
131
132 return delay;
133}
