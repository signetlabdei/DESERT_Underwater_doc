181 virtual void
182 freeze()
183 {
184 assert(timer_status == UWPOLLING_RUNNING);
185 left_duration -= (NOW - start_time);
186 if (left_duration <= 0.0) {
187 left_duration = module->mac2phy_delay_;
188 }
189 force_cancel();
190 timer_status = UWPOLLING_FROZEN;
191 }
192
197 virtual void
198 unFreeze()
199 {
200 assert(timer_status == UWPOLLING_FROZEN);
201 start_time = NOW;
202 assert(left_duration > 0);
203 sched(left_duration);
204 timer_status = UWPOLLING_RUNNING;
205 }
206
210 virtual void
211 stop()
212 {
213 timer_status = UWPOLLING_IDLE;
214 force_cancel();
215 }
216
221 virtual void
222 schedule(double val)
223 {
224 start_time = NOW;
225 left_duration = val;
226 timer_status = UWPOLLING_RUNNING;
227 resched(val);
228 }
