58 bool Start(bool exc_info = false)
59 {
60 /* Check for double-start */
61 if (Running())
62 return false;
63 try {
64 m_thread = std::thread(&StoppableThread::RunInternal, this);
65 return true;
66 }
67 catch (std::exception& ex) {
68 if (exc_info)
69 std::cerr << "StoppableThread::Start() exception: " << ex.what() << std::endl;
70 }
71 catch (...) {
72 if (exc_info)

74 }

76 }

83 {
