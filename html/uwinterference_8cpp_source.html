107 void
108 uwinterference::addToInterference(double pw, PKT_TYPE tp)
109 {
110  if (use_maxinterval_) {
111  std::list<ListNode>::iterator it;
112  for (it = power_list.begin(); it != power_list.end();) {
113  if (it->time < NOW - maxinterval_) {
114  it = power_list.erase(it); // Side effect: it++
115  } else
116  break;
117  }
118  }
119 
120  if (power_list.empty()) {
121  if (tp == CTRL) {
122  ListNode pn(NOW, pw, 1, 0);
123  power_list.push_back(pn);
124  } else {
125  ListNode pn(NOW, pw, 0, 1);
126  power_list.push_back(pn);
127  }
128  } else {
129  double power_temp = power_list.back().sum_power;
130  int ctrl_temp = power_list.back().ctrl_cnt;
131  int data_temp = power_list.back().data_cnt;
132  if (tp == CTRL) {
133  ListNode pn(NOW, pw + power_temp, ctrl_temp + 1, data_temp);
134  power_list.push_back(pn);
135  } else {
136  ListNode pn(NOW, pw + power_temp, ctrl_temp, data_temp + 1);
137  power_list.push_back(pn);
138  }
139  }
140 
141  if (debug_) {
142  std::cout << NOW << " uwinterference::addToInterference, power: " << pw
143  << " ,total power: " << power_list.back().sum_power
144  << " ,ctrl_packet: " << power_list.back().ctrl_cnt
145  << " ,data packets: " << power_list.back().data_cnt
146  << std::endl;
147  }

150 void
151 uwinterference::removeFromInterference(double pw, PKT_TYPE tp)
152 {
153  if (use_maxinterval_) {
154  std::list<ListNode>::iterator it;
155 
156  for (it = power_list.begin(); it != power_list.end();) {
157  if (it->time < NOW - maxinterval_) {
158  it = power_list.erase(it); // Side effect: it++
159  } else
160  break;
161  }
162  }
163 
164  if (power_list.empty()) {
165  std::cerr << "uwinterference::removeFromInterference, "
166  << "some interference removed wrongly" << std::endl;
167  } else {
168  double power_temp = power_list.back().sum_power;
169  int ctrl_temp = power_list.back().ctrl_cnt;
170  int data_temp = power_list.back().data_cnt;
171 
172  if (tp == CTRL) {
173  // NOW+EPSILON_TIME to compensate the early scheduling in
174  // addToInterference(Packet* p)
175  ListNode pn(NOW + EPSILON_TIME,
176  power_temp - pw,
177  ctrl_temp - 1,
178  data_temp);
179  power_list.push_back(pn);
180  } else {
181  // NOW+EPSILON_TIME to compensate the early scheduling in
182  // addToInterference(Packet* p)
183  ListNode pn(NOW + EPSILON_TIME,
184  power_temp - pw,
185  ctrl_temp,
186  data_temp - 1);
187  power_list.push_back(pn);
188  }
189  }
190  if (debug_) {
191  std::cout << NOW << " uwinterference::removeFromInterference, "
192  << "power: " << pw
193  << " ,total power: " << power_list.back().sum_power
194  << " ,ctrl_packet: " << power_list.back().ctrl_cnt
195  << " ,data packets: " << power_list.back().data_cnt
196  << std::endl;
197  }
198 
199  double power_temp = power_list.back().sum_power;
200  if (power_temp < 0) {
201  if (debug_) {
202  std::cout << NOW
203  << " Precision error, negative power: " << power_temp
204  << std::endl;
205  }
206  power_list.back().sum_power = 0;
207  }

223 double
224 uwinterference::getInterferencePower(
225  double power, double starttime, double duration)
226 {
227  std::list<ListNode>::reverse_iterator rit;
228 
229  double integral = 0;
230  double lasttime = NOW;
231  assert(starttime <= NOW);
232  assert(duration > 0);
233 
234  for (rit = power_list.rbegin(); rit != power_list.rend(); ++rit) {
235  if (starttime < rit->time) {
236  integral += rit->sum_power * (lasttime - rit->time);
237  lasttime = rit->time;
238 
239  } else {
240  integral += rit->sum_power * (lasttime - starttime);
241 
242  break;
243  }
244  }
245  double interference = (integral / duration) - power;
246 
247  if (abs(interference) < POWER_PRECISION_THRESHOLD) {
248  if (debug_)
249  std::cout << "getInterferencePower() WARNING:"
250  << " interference=" << interference
251  << " POWER_PRECISION_THRESHOLD="
252  << POWER_PRECISION_THRESHOLD
253  << ". Precision error, interference set to 0"
254  << endl;
255  }
256  // Check for cancellation errors
257  // which can arise when interference is subtracted
258  if (interference < 0) {
259  interference = 0;
260  if (debug_)
261  std::cout << "getInterferencePower() WARNING:"
262  << " cancellation errors, interference set to 0"
263  << endl;
264  }
265 
266  if (debug_) {
267  std::cout << "transmission from " << starttime << " to "
268  << starttime + duration << " power " << power
269  << " gets interference " << interference << std::endl;
270  }
271  return interference;
272 }

291 double
292 uwinterference::getTimeOverlap(double starttime, double duration)
293 {
294  std::list<ListNode>::reverse_iterator rit;
295 
296  double overlap = 0;
297  double lasttime = NOW;
298  assert(starttime <= NOW);
299  assert(duration > 0);
300 
301  for (rit = power_list.rbegin(); rit != power_list.rend(); ++rit) {
302  if (starttime < rit->time) {
303 
304  if (rit->ctrl_cnt > 1 || rit->data_cnt > 1) {
305  overlap += (lasttime - rit->time);
306  }
307  lasttime = rit->time;
308 
309  } else {
310  if (rit->ctrl_cnt > 1 || rit->data_cnt > 1) {
311  overlap += (lasttime - starttime);
312  }
313  break;
314  }
315  }
316 
317  return overlap / duration;
318 }

334 counter
335 uwinterference::getCounters(double starttime, double duration, PKT_TYPE tp)
336 {
337  std::list<ListNode>::reverse_iterator rit;
338 
339  int ctrl_pkts = 0;
340  int data_pkts = 0;
341 
342  assert(starttime <= NOW);
343  assert(duration > 0);
344 
345  rit = power_list.rbegin();
346  int last_ctrl_cnt = rit->ctrl_cnt;
347  int last_data_cnt = rit->data_cnt;
348  rit++;
349  for (; rit != power_list.rend(); ++rit) {
350  if (starttime < rit->time) {
351  if (last_ctrl_cnt - rit->ctrl_cnt >= 0) {
352  ctrl_pkts += last_ctrl_cnt - rit->ctrl_cnt;
353  }
354  if (last_data_cnt - rit->data_cnt >= 0) {
355  data_pkts += last_data_cnt - rit->data_cnt;
356  }
357  last_ctrl_cnt = rit->ctrl_cnt;
358  last_data_cnt = rit->data_cnt;
359  } else {
360  ctrl_pkts += rit->ctrl_cnt;
361  data_pkts += rit->data_cnt;
362  break;
363  }
364  }
365 
366  if (tp == CTRL) {
367  ctrl_pkts--;
368  } else {
369  data_pkts--;
370  }
371 
372  if (debug_) {
373  std::cout << NOW << " uwinterference::getCounters(), collisions"
374  << " with ctrl pkts: " << ctrl_pkts
375  << ", collisions with data pkts: " << data_pkts << std::endl;
376  }
377 
378  return counter(ctrl_pkts, data_pkts);
