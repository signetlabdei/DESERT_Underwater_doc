121extern "C" int Uwmulti_traffic_control_Init()
122{
123 PT_MULTI_TR_PROBE = p_info::addPacket("MULTI_TR_PROBE");
124 PT_MULTI_TR_PROBE_ACK = p_info::addPacket("MULTI_TR_PROBE_ACK");
125
126 PT_UWMULTIPHY_PING = p_info::addPacket((char*) "UWMULTIPHY/UWMULTIPHY_PING");
127 PT_UWMULTIPHY_PONG = p_info::addPacket((char*) "UWMULTIPHY/UWMULTIPHY_PONG");
128
129 PT_UWMULTIPHY_DATA = p_info::addPacket((char*) "UWMULTIPHY/UWMULTIPHY_DATA");
130
131 UwMultiTrafficControlInitTclCode.load();
132 return 0;
133}
