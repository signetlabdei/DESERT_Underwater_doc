79  hdr_mac *mach = HDR_MAC(p);
80 
81  if (ch->ptype() == PT_TLOHI) {
82  hdr_tlohi *tlohih = HDR_TLOHI(p);
83 
84  writeTrace(sap,
85  " MACSRC=%d MACDST=%d SN=%d TS=%f TP=%s SZ=%d",
86  mach->macSA(),
87  mach->macDA(),
88  tlohih->sn,
89  tlohih->ts,
90  printType(tlohih->pkt_type),
91  ch->size());
92  }
93 
94  else if (ch->ptype() == PT_WKUP) {
95  writeTrace(sap, " TONE");
96  }

98  else
