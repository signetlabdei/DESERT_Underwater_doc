230 modem_state_t
231 MdriverS2C_EvoLogics::updateStatus()
232 {
233 
234  std::stringstream sstr("");
235  string strlog;
236  sstr << "UPDATE_STATUS::STATUS_" << status << "_M_STATUS_TX_" << m_status_tx
237  << "_M_STATUS_RX_" << m_status_rx;
238  sstr >> strlog;
239  printOnLog(LOG_LEVEL_INFO, "MS2C_EVOLOGICSDRIVER", strlog);
240 
241  std::string rx_msg;
242  bool cread = true;
243 
244  while (cread) {
245  rx_msg = mConnector.readFromModem();
246  if (rx_msg != "") {
247  std::string pr_msg;
248  std::string parser("\r\n");
249  size_t p_offset = 0;
250  size_t p_parser = rx_msg.find(parser);
251 
252  while (p_parser != std::string::npos) {
253  pr_msg = rx_msg.substr(
254  p_offset, p_parser + parser.size() - p_offset);
255  if (status == MODEM_MULTIPATH) {
256 
257  status = MODEM_IDLE;
258  m_status_tx = TX_STATE_IDLE;
259  cread = false;
260  break;
261  }
262  if ((pr_msg.find("RECVIM") != string::npos) ||
263  (pr_msg.find("RECV") != string::npos)) {
264  if (!getResetModemQueue()) {
265  queue_rx.push(pr_msg);
266  }
267  } else if ((pr_msg.find("OK") != string::npos) ||
268  (pr_msg.find("BUSY CLOSING CONNECTION") !=
269  string::npos) ||
270  (pr_msg.find("BUSY BACKOFF STATE") != string::npos) ||
271  (pr_msg.find("BUSY DELIVERING") != string::npos) ||
272  (pr_msg.find("FAILEDIM") != string::npos) ||
273  (pr_msg.find("FAILED") != string::npos) ||
274  (pr_msg.find("ERROR") != string::npos) ||
275  (pr_msg.find("INITIATION NOISE") != string::npos) ||
276  (pr_msg.find("INITIATION LISTEN") != string::npos) ||
277  (pr_msg.at(0) == '-' || std::isdigit(pr_msg.at(0))) ||
278  (pr_msg.find("[*]OK") != string::npos) ||
279  (is_number(pr_msg))) {
280  queue_tx.push(pr_msg);
281  }
282  p_offset += pr_msg.size();
283  p_parser = rx_msg.find(parser, p_offset + 1);
284  } // End while (p_parser!=std::string::npos)
285 
286  } else {
287  cread = false; // exit the loop
288  } // End if (rx_msg!="")
289  } // End of while (cread)
290  cread = true;
291  if (status == MODEM_TX || status == MODEM_CFG || status == MODEM_RESET ||
292  status == MODEM_QUIT || status == MODEM_NOISE ||
293  status == MODEM_MULTIPATH || status == MODEM_CHANGE_POWER_LEVEL) {
294  if (m_status_rx == RX_STATE_RX_IM && status == MODEM_TX) {
295  // Update S2C RX status
296  m_status_rx = RX_STATE_IDLE;
297  }
298 
299  while (cread) {
300 
301  if (!queue_tx.empty()) {
302 
303  // Read the possible received message
304  rx_msg = queue_tx.front();
305  queue_tx.pop();
306 
307  if (rx_msg.find("OK") != string::npos ||
308  (rx_msg.find("[*]OK") != string::npos)) {
309  std::stringstream sstr("");
310  string strlog;
311  sstr << "UPDATE_STATUS::PROCESSING_OK::" << status
312  << "_M_STATUS_TX_" << m_status_tx << "_M_STATUS_RX_"
313  << m_status_rx;
314  sstr >> strlog;
315  printOnLog(LOG_LEVEL_INFO, "MS2C_EVOLOGICSDRIVER", strlog);
316  if (status == MODEM_RESET || status == MODEM_CFG ||
317  status == MODEM_QUIT) {
318 
319  if (getKeepOnlineMode()) {
320  status = MODEM_CFG;
321  m_status_tx = TX_STATE_SET_KO;
322  cread = false;
323  } else if (getModemID()) {
324  status = MODEM_CFG;
325  m_status_tx = TX_STATE_SET_ID;
326  cread = false;
327  } else if (clmsg_c_power_level) {
328  printOnLog(LOG_LEVEL_INFO,
329  "MS2C_EVOLOGICSDRIVER",
330  "REQUESTED_POWER_CHANGE");
331  status = MODEM_CHANGE_POWER_LEVEL;
332  cread = false;
333  } else {
334  status = MODEM_IDLE;
335  m_status_tx = TX_STATE_IDLE;
336  cread = false;
337  }
338 
339  } else if (status == MODEM_TX) {
340  tx_packet_counter_noise++;
341  tx_packet_counter_multipath++;
342  if (clmsg_c_power_level) {
343  printOnLog(LOG_LEVEL_INFO,
344  "MS2C_EVOLOGICSDRIVER",
345  "REQUESTED_POWER_CHANGE");
346  status = MODEM_CHANGE_POWER_LEVEL;
347  m_status_tx = TX_STATE_SEND_ATL;
348  } else if (static_cast<MS2C_EvoLogics *>(pmModem)
349  ->getNoiseProbeFrequency() > 0) {
350  if (tx_packet_counter_noise >=
351  static_cast<MS2C_EvoLogics *>(pmModem)
352  ->getNoiseProbeFrequency()) {
353  printOnLog(LOG_LEVEL_INFO,
354  "MS2C_EVOLOGICSDRIVER",
355  "PROBING_MODEM_NOISE");
356  tx_packet_counter_noise = 0;
357  status = MODEM_NOISE;
358  m_status_tx = TX_STATE_SEND_ATN;
359  usleep(1000000);
360  }
361  } else if (static_cast<MS2C_EvoLogics *>(pmModem)
362  ->getMultipathProbeFrequency() > 0) {
363  if (tx_packet_counter_multipath >=
364  static_cast<MS2C_EvoLogics *>(pmModem)
365  ->getMultipathProbeFrequency()) {
366  printOnLog(LOG_LEVEL_INFO,
367  "MS2C_EVOLOGICSDRIVER",
368  "PROBING_MODEM_MULTIPATH");
369  tx_packet_counter_multipath = 0;
370  status = MODEM_MULTIPATH;
371  m_status_tx = TX_STATE_SEND_ATP;
372  } else {
373  status = MODEM_IDLE;
374  m_status_tx = TX_STATE_IDLE;
375  }
376  } else {
377  status = MODEM_IDLE;
378  m_status_tx = TX_STATE_IDLE;
379  }
380  cread = false;
381  } else if (status == MODEM_CHANGE_POWER_LEVEL) {
382  if (tx_packet_counter_noise >=
383  static_cast<MS2C_EvoLogics *>(pmModem)
384  ->getNoiseProbeFrequency() &&
385  static_cast<MS2C_EvoLogics *>(
386  pmModem)->getNoiseProbeFrequency() >
387  0) {
388  printOnLog(LOG_LEVEL_INFO,
389  "MS2C_EVOLOGICSDRIVER",
390  "PROBING_MODEM_NOISE");
391  tx_packet_counter_noise = 0;
392  status = MODEM_NOISE;
393  m_status_tx = TX_STATE_SEND_ATN;
394  usleep(1000000);
395  } else if (static_cast<MS2C_EvoLogics *>(pmModem)
396  ->getMultipathProbeFrequency() > 0) {
397  if (tx_packet_counter_multipath >=
398  static_cast<MS2C_EvoLogics *>(pmModem)
399  ->getMultipathProbeFrequency()) {
400  printOnLog(LOG_LEVEL_INFO,
401  "MS2C_EVOLOGICSDRIVER",
402  "PROBING_MODEM_MULTIPATH");
403  tx_packet_counter_multipath = 0;
404  status = MODEM_MULTIPATH;
405  m_status_tx = TX_STATE_SEND_ATP;
406  } else {
407  status = MODEM_IDLE;
408  m_status_tx = TX_STATE_IDLE;
409  }
410  } else {
411  status = MODEM_IDLE;
412  m_status_tx = TX_STATE_IDLE;
413  }
414  cread = false;
415  } else {
416 
417  std::stringstream sstr("");
418  string strlog;
419  sstr << "UPDATE_STATUS::OK_WRONG_STATUS_" << m_status_tx
420  << "_" << status;
421  sstr >> strlog;
422  printOnLog(
423  LOG_LEVEL_INFO, "MS2C_EVOLOGICSDRIVER", strlog);
424  }
425 
426  } else if (rx_msg.find("ERROR") != string::npos) {
427  if (status == MODEM_CFG || status == MODEM_TX ||
428  status == MODEM_RESET ||
429  status == MODEM_CHANGE_POWER_LEVEL) {
430 
431  std::stringstream sstr("");
432  string strlog;
433  sstr << "UPDATE_STATUS::PROCESSING_ERROR_"
434  << m_status_tx << "_" << status;
435  sstr >> strlog;
436  printOnLog(LOG_LEVEL_ERROR,
437  "MS2C_EVOLOGICSDRIVER",
438  strlog);
439  status = MODEM_IDLE;
440  } else {
441  std::stringstream sstr("");
442  string strlog;
443  sstr << "UPDATE_STATUS::ERROR_WRONG_STATUS_TX_"
444  << m_status_tx << "_STATUS_" << status;
445  sstr >> strlog;
446  printOnLog(LOG_LEVEL_ERROR,
447  "MS2C_EVOLOGICSDRIVER",
448  strlog);
449  }
450  m_status_tx = TX_STATE_IDLE;
451  cread = false;
452 
453  } else if (rx_msg.find("BUSY DELIVERING") != string::npos) {
454  printOnLog(LOG_LEVEL_ERROR,
455  "MS2C_EVOLOGICSDRIVER",
456  "UPDATE_STATUS::BUSY_DELIVERING_MESSAGE");
457  status = MODEM_IDLE;
458  m_status_tx = TX_STATE_IDLE;
459  cread = false;
460  } else if (rx_msg.find("INITIATION NOISE") != string::npos) {
461  printOnLog(LOG_LEVEL_ERROR,
462  "MS2C_EVOLOGICSDRIVER",
463  "UPDATE_STATUS::INITIATION_NOISE");
464  status = MODEM_NOISE;
465  m_status_tx = TX_STATE_SEND_ATE;
466  cread = false;
467  } else if (rx_msg.at(0) == '-' || is_number(rx_msg)) {
468  std::stringstream sstr("");
469  string strlog;
470  sstr << "UPDATE_STATUS::NOISE_VALUE_RECEIVED_" << rx_msg;
471  sstr >> strlog;
472  printOnLog(LOG_LEVEL_ERROR, "MS2C_EVOLOGICSDRIVER", strlog);
473  status = MODEM_IDLE;
474  m_status_tx = TX_STATE_SEND_ATA;
475  cread = false;
476  } else if (rx_msg.find("INITIATION LISTEN") != string::npos) {
477  status = MODEM_IDLE;
478  m_status_tx = TX_STATE_IDLE;
479  if (clmsg_c_power_level) {
480  status = MODEM_CHANGE_POWER_LEVEL;
481  }
482  cread = false;
483  } else {
484 
485  status = MODEM_IDLE;
486  }
487  } else {
488  cread = false;
489  }
490  }
491  } else if (status == MODEM_RX) {
492 
493  m_status_rx = RX_STATE_IDLE;
494  status = MODEM_IDLE_RX;
495  cread = false;
496 
497  } else { // read from queue_rx only
498 
499  while (cread) {
500  if (!queue_rx.empty()) {
501  // Read the possible received message
502  rx_msg = queue_rx.front();
503  queue_rx.pop();
504 
505  if (rx_msg.find("RECVIM") != string::npos) {
506  m_status_rx = RX_STATE_RX_IM;
507  status = MODEM_RX;
508  mInterpreter.parse_recvim(rx_msg);
509  cread = false;
510  } else if (rx_msg.find("RECV") != string::npos) {
511  m_status_rx = RX_STATE_RX_BURST;
512  status = MODEM_RX;
513  mInterpreter.parse_recv(rx_msg);
514  cread = false;
515  } else if (rx_msg.find("RECVPBM") != string::npos) {
516  m_status_rx = RX_STATE_RX_BURST;
517  status = MODEM_RX;
518  mInterpreter.parse_recvpbm(rx_msg);
519  cread = false;
520  } else { // Any other AT messages (ignored for the moment)
521  std::stringstream sstr("");
522  string strlog;
523  sstr << "UPDATE_STATUS::IGNORED_AT_MESSAGE_" << rx_msg;
524  sstr >> strlog;
525  printOnLog(LOG_LEVEL_ERROR, "MS2C_EVOLOGICSDRIVER", strlog);
526  }
527  } else {
528  cread = false;
529  } // End if (!queue_rx.empty())
530  } // End while (cread)
531 
532  } // End if (status == _TX || status == _CFG)
533  return status;

536 void
537 MdriverS2C_EvoLogics::modemTxManager()
538 {
539  std::string tx_msg;
540  std::stringstream sstr("");
541  string strlog;
542  switch (m_status_tx) {
543  case TX_STATE_SET_ID:
544  // Build the configuration message
545  tx_msg = mInterpreter.build_setAL(ID);
546  sstr << "MODEM_TX_MANAGER::SETAL!" << ID << "_MESSAGE";
547  sstr >> strlog;
548  printOnLog(LOG_LEVEL_INFO, "MS2C_EVOLOGICSDRIVER", strlog);
549  break;
550  case TX_STATE_SEND_IM:
551  // Build the instant message
552  tx_msg = mInterpreter.build_sendim(
553  payload_tx.length(), dest, "noack", payload_tx);
554  sstr << "MODEM_TX_MANAGER::SENDIM = " << hexdumplog(tx_msg);
555  sstr >> strlog;
556  printOnLog(LOG_LEVEL_INFO, "MS2C_EVOLOGICSDRIVER", strlog);
557  break;
558  case TX_STATE_SEND_BURST:
559  // build a burst data
560  tx_msg = mInterpreter.build_atsend(
561  payload_tx.length(), dest, payload_tx);
562  sstr << "MODEM_TX_MANAGER::SEND = " << hexdumplog(tx_msg);
563  sstr >> strlog;
564  printOnLog(LOG_LEVEL_INFO, "MS2C_EVOLOGICSDRIVER", strlog);
565  break;
566  case TX_STATE_SEND_PBM:
567  tx_msg = mInterpreter.build_atsendpbm(
568  payload_tx.length(), dest, payload_tx);
569  sstr << "MODEM_TX_MANAGER::SEND_PBM = " << hexdumplog(tx_msg);
570  sstr >> strlog;
571  printOnLog(LOG_LEVEL_INFO, "MS2C_EVOLOGICSDRIVER", strlog);
572  break;
573  case TX_STATE_DROPBUFFER:
574  tx_msg = mInterpreter.build_atzn(DROPBUFFER_TYPE);
575  break;
576  case TX_STATE_SET_KO:
577  sstr << "MODEM_TX_MANAGER::SEND_AT!KO = " << hexdumplog(tx_msg);
578  sstr >> strlog;
579  printOnLog(LOG_LEVEL_INFO, "MS2C_EVOLOGICSDRIVER", strlog);
580  tx_msg = mInterpreter.build_atko(0);
581  break;
582  case TX_STATE_SEND_CLOSE:
583  tx_msg = mInterpreter.build_ath(0);
584  break;
585  case TX_STATE_SEND_ATN:
586  tx_msg = mInterpreter.build_atn();
587  break;
588  case TX_STATE_SEND_ATE:
589  tx_msg = mInterpreter.build_ate();
590  break;
591  case TX_STATE_SEND_ATA:
592  tx_msg = mInterpreter.build_ata();
593  break;
594  case TX_STATE_SEND_ATP:
595  tx_msg = mInterpreter.build_atp();
596  break;
597  case TX_STATE_SEND_ATL:
598  tx_msg = mInterpreter.build_atl(actual_power_level);
599  break;
600  default:
601 
602  sstr << "MODEM_TX_MANAGER::UNEXPECTED_CALL_m_status_tx = "
603  << m_status_tx;
604  sstr >> strlog;
605  printOnLog(LOG_LEVEL_ERROR, "MS2C_EVOLOGICSDRIVER", strlog);
606  return;
607  } // end switch
608  mConnector.writeToModem(tx_msg);
609 }
