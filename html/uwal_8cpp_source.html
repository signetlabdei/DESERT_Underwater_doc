103 int
104 Uwal::command(int argc, const char *const *argv)
105 {
106  if (argc == 2) {
107  if (strcmp(argv[1], "Reset_PER_List") == 0) {
108  PERList.clear();
109  return TCL_OK;
110  }
111  }
112  if (argc == 3) {
113  if (strcmp(argv[1], "linkPacker") == 0) { // tcl command to link to this
114  // Uwal object a packer object
115  // shadowed as an OTcl object.
116  pPacker = (packer *) TclObject::lookup(argv[2]);
117  return TCL_OK;
118  }
119 
120  if (strcmp(argv[1], "setDummyStr") ==
121  0) { // tcl command to set a customized dummy string.
122  dummyStr = "";
123  while (dummyStr.size() < PSDU) {
124  dummyStr += argv[2];
125  }
126  return TCL_OK;
127  }
128  }
129  if (argc >= 3) {
130  if (strcmp(argv[1], "Set_PER_List") == 0) {
131  for (int i = 2; i < argc - 1; i = i + 2) {
132  int id = atoi(argv[i]);
133  double per = atof(argv[i + 1]);
134  pl_element pl;
135  pl.node_ID = id;
136  pl.per = per;
137  for (list<pl_element>::iterator it = PERList.begin();
138  it != PERList.end();
139  ++it) {
140  if (it->node_ID == pl.node_ID) {
141  it = PERList.erase(it);
142  }
143  }
144  PERList.insert(PERList.begin(), pl);
145  }
146  return TCL_OK;
147  }
148  if (strcmp(argv[1], "Clear_PER_List") == 0) {
149  for (int i = 2; i < argc; i++) {
150  int id = atoi(argv[i]);
151  for (list<pl_element>::iterator it = PERList.begin();
152  it != PERList.end();
153  ++it) {
154  if (it->node_ID == id) {
155  it = PERList.erase(it);
156  }
157  }
158  }
159  return TCL_OK;
160  }
161  }
162  return Module::command(argc, argv);
163 }

234 {
235 
236  if (!sendDownFrames.empty()) {
237  startTx(sendDownFrames.front());

288 void
289 Uwal::fragmentPkt(Packet *p)
290 {
291 
292  hdr_uwal *hal = HDR_UWAL(p);
293  hdr_cmn *ch = HDR_CMN(p);
294 
295  if (pPacker == NULL) {
296  if (debug_) {
297  printf("\033[0;0;31m WARNING: \033[0m ");
298  cout << "in Uwal::fragmentPkt -> pPacker set to NULL (i.e., no UW "
299  "AL packer added from tcl with linkPacker). No "
300  "serialization neither fragmentation can be performed! "
301  "(i.e., packet just sent to lower layers)."
302  << endl;
303  }
304  sendDownFrames.push(p);
305  } else {
306  size_t hdr_length = pPacker->getHdrBytesLength();
307  // size_t payload_length = pPacker->getPayloadBytesLength();
308  // size_t pkt_length = hdr_length + payload_length;
309  size_t pkt_length = hal->binPktLength();
310  size_t payload_length =
311  pkt_length - hdr_length; // pPacker->getPayloadBytesLength();
312 
313  if (PSDU < hdr_length) {
314  printf("\033[0;0;31m ERROR: \033[0m ");
315  cout << "in packer::ALfragmenter PSDU: " << PSDU
316  << " but hdr_length; " << hdr_length
317  << ". Packet fragmentation is not possible since the hdr "
318  "cannot fit in the PSDU."
319  << endl;
320  return;
321  }
322 
323  if (pkt_length <= PSDU) { // !(PSDU < pkt_length)) {
324  if (frame_padding) {
325  hal->binPktLength() = PSDU;
326  ch->size_ = PSDU;
327  }
328  sendDownFrames.push(p);
329  } else {
330  Packet *f_tmp;
331  hdr_uwal *hal_tmp;
332  hdr_cmn *ch_tmp;
333 
334  size_t framePayloadLength = PSDU - hdr_length;
335  size_t frameNumber = payload_length / framePayloadLength;
336  // size_t frameNumber = HDR_UWAL(p)->binPktLength() /
337  // framePayloadLength;
338 
339  size_t lastFramePayloadLength = hal->binPktLength() -
340  hal->binHdrLength() - frameNumber * framePayloadLength;
341 
342  if (debug_) {
343  std::cout << "------ ch->size_: " << ch->size()
344  << " hdr_length: " << hdr_length
345  << ", payload_length: " << payload_length
346  << ", pkt_length: " << pkt_length
347  << ", framePayloadLength: " << framePayloadLength
348  << ", frameNumber: " << frameNumber
349  << " lastFramePayloadLength: "
350  << lastFramePayloadLength << std::endl;
351  }
352 
353  for (uint i = 0; i < frameNumber; i++) {
354  f_tmp = p->copy();
355  initializeHdr(f_tmp, hal->pktID());
356 
357  hal_tmp = HDR_UWAL(f_tmp);
358  ch_tmp = HDR_CMN(f_tmp);
359 
360  // hal_tmp->framePayloadOffset() = i*framePayloadLength;
361  hal_tmp->framePayloadOffset() = i;
362  if (!(i == (frameNumber - 1) && lastFramePayloadLength == 0))
363  hal_tmp->Mbit() = 1;
364 
365  pPacker->packHdr(f_tmp);
366 
367  // memcpy(hal_tmp->binPkt() + hal_tmp->binHdrLength(),
368  // hal->binPkt() + hal->binHdrLength() +
369  // hal_tmp->framePayloadOffset(), framePayloadLength);
370  memcpy(hal_tmp->binPkt() + hal_tmp->binHdrLength(),
371  hal->binPkt() + hal->binHdrLength() +
372  hal_tmp->framePayloadOffset() *
373  framePayloadLength,
374  framePayloadLength);
375  hal_tmp->binPktLength() += framePayloadLength;
376 
377  if (debug_) {
378  if (i == (frameNumber - 1) && lastFramePayloadLength == 0)
379  std::cout << "TX frame (last) num: " << i << endl;
380  else
381  std::cout << "TX frame num: " << i << endl;
382  std::cout << "Header: "
383  << pPacker->hexdump(hal_tmp->binPkt(),
384  hal_tmp->binHdrLength())
385  << endl;
386  std::cout << "Payload: "
387  << pPacker->hexdump(hal_tmp->binPkt() +
388  hal_tmp->binHdrLength(),
389  hal_tmp->binPktLength() -
390  hal_tmp->binHdrLength())
391  << endl;
392  }
393 
394  ch_tmp->size_ = PSDU;
395  sendDownFrames.push(f_tmp);
396  }
397 
398  if (lastFramePayloadLength > 0) {
399 
400  f_tmp = p->copy();
401  initializeHdr(f_tmp, hal->pktID());
402 
403  hal_tmp = HDR_UWAL(f_tmp);
404  ch_tmp = HDR_CMN(f_tmp);
405  // hal_tmp->framePayloadOffset() =
406  // frameNumber*framePayloadLength;
407  hal_tmp->framePayloadOffset() = frameNumber;
408 
409  pPacker->packHdr(f_tmp);
410 
411  if (frame_padding) {
412  memcpy(hal_tmp->binPkt() + hal_tmp->binHdrLength(),
413  hal->binPkt() + hal->binHdrLength() +
414  hal_tmp->framePayloadOffset() *
415  framePayloadLength,
416  PSDU - hal_tmp->binHdrLength());
417  hal_tmp->binPktLength() += (PSDU - hal_tmp->binHdrLength());
418  ch_tmp->size_ = PSDU;
419  } else {
420  memcpy(hal_tmp->binPkt() + hal_tmp->binHdrLength(),
421  hal->binPkt() + hal->binHdrLength() +
422  hal_tmp->framePayloadOffset() *
423  framePayloadLength,
424  lastFramePayloadLength);
425  hal_tmp->binPktLength() += lastFramePayloadLength;
426  ch_tmp->size_ =
427  hal_tmp->binHdrLength() + lastFramePayloadLength;
428  }
429 
430  if (debug_) {
431  std::cout << "TX (last) frame num: " << frameNumber << endl;
432  std::cout << "Header: "
433  << pPacker->hexdump(hal_tmp->binPkt(),
434  hal_tmp->binHdrLength())
435  << endl;
436  std::cout << "Payload: "
437  << pPacker->hexdump(hal_tmp->binPkt() +
438  hal_tmp->binHdrLength(),
439  hal_tmp->binPktLength() -
440  hal_tmp->binHdrLength())
441  << endl;
442  }
443 
444  sendDownFrames.push(f_tmp);
445  }
446 
447  Packet::free(p);
448  }
449  }

452 void
453 Uwal::reassembleFrames(Packet *p)
454 {
455  hdr_uwal *hal = HDR_UWAL(p);
456  hdr_cmn *ch = HDR_CMN(p);
457 
458  if (pPacker == NULL) {
459  if (debug_) {
460  printf("\033[0;0;31m WARNING: \033[0m ");
461  cout << "in Uwal::fragmentPkt -> pPacker set to NULL (i.e., no UW "
462  "AL packer added from tcl with linkPacker). No "
463  "serialization neither reassembling of fragmented packets "
464  "can be performed! (i.e., frame just sent to upper layers)."
465  << endl;
466  }
467  sendUpPkts.push(p);
468  } else {
469  RxFrameSetKey newKey(hal->srcID(), hal->pktID());
470  std::map<RxFrameSetKey, RxFrameSet>::iterator it =
471  sendUpFrameSet.find(newKey);
472 
473  size_t framePayloadOffset =
474  hal->framePayloadOffset() * (PSDU - hal->binHdrLength());
475  // size_t framePayloadOffset = hal->framePayloadOffset()*(ch->size() -
476  // hal->binHdrLength());
477 
478  if (it != sendUpFrameSet.end()) {
479  if (hal->Mbit()) { // not the last pkt frame
480  //(it->second).UpdateRxFrameSet(hal->binPkt() +
481  // hal->binHdrLength(), hal->framePayloadOffset(),
482  // hal->binPktLength() - hal->binHdrLength(), -1,
483  // Scheduler::instance().clock());
484  (it->second)
485  .UpdateRxFrameSet(hal->binPkt() + hal->binHdrLength(),
486  framePayloadOffset,
487  hal->binPktLength() - hal->binHdrLength(),
488  -1,
489  Scheduler::instance().clock());
490  } else {
491  //(it->second).UpdateRxFrameSet(hal->binPkt() +
492  // hal->binHdrLength(), hal->framePayloadOffset(),
493  // hal->binPktLength() - hal->binHdrLength(),
494  //(hal->framePayloadOffset() + hal->binPktLength() -
495  // hal->binHdrLength()), Scheduler::instance().clock());
496  (it->second)
497  .UpdateRxFrameSet(hal->binPkt() + hal->binHdrLength(),
498  framePayloadOffset,
499  hal->binPktLength() - hal->binHdrLength(),
500  (framePayloadOffset + hal->binPktLength() -
501  hal->binHdrLength()),
502  Scheduler::instance().clock());
503  }
504 
505  if (ch->error()) {
506  if (debug_) {
507  std::cout << NOW << " UW-AL(" << nodeID
508  << ") - Received frame in error" << std::endl;
509  }
510  (it->second).setError();
511  }
512 
513  } else {
514  RxFrameSet newSet;
515 
516  if (hal->Mbit()) { // not the unique or last pkt frame
517  // newSet.UpdateRxFrameSet(hal->binPkt() + hal->binHdrLength(),
518  // hal->framePayloadOffset(), hal->binPktLength() -
519  // hal->binHdrLength(), -1, Scheduler::instance().clock());
520  newSet.UpdateRxFrameSet(hal->binPkt() + hal->binHdrLength(),
521  framePayloadOffset,
522  hal->binPktLength() - hal->binHdrLength(),
523  -1,
524  Scheduler::instance().clock());
525  } else {
526  // newSet.UpdateRxFrameSet(hal->binPkt() + hal->binHdrLength(),
527  // hal->framePayloadOffset(), hal->binPktLength() -
528  // hal->binHdrLength(), (hal->framePayloadOffset() +
529  // hal->binPktLength() - hal->binHdrLength()),
530  // Scheduler::instance().clock());
531  newSet.UpdateRxFrameSet(hal->binPkt() + hal->binHdrLength(),
532  framePayloadOffset,
533  hal->binPktLength() - hal->binHdrLength(),
534  (hal->framePayloadOffset() + hal->binPktLength() -
535  hal->binHdrLength()),
536  Scheduler::instance().clock());
537  }
538 
539  if (ch->error()) {
540  if (debug_) {
541  std::cout << NOW << " UW-AL(" << nodeID
542  << ") - Received frame in error" << std::endl;
543  }
544  newSet.setError();
545  }
546 
547  sendUpFrameSet.insert(
548  std::pair<RxFrameSetKey, RxFrameSet>(newKey, newSet));
549  }
550 
551  Packet::free(p);
552 
553  if (debug_) {
554  std::cout << NOW << " UW-AL(" << nodeID
555  << ") Generated map of RxFrameSets. Number of elements: "
556  << sendUpFrameSet.size() << endl;
557  int i = 1;
558  for (std::map<RxFrameSetKey, RxFrameSet>::iterator it =
559  sendUpFrameSet.begin();
560  it != sendUpFrameSet.end();
561  it++) {
562  std::cout << "Element num: " << i++ << endl;
563  std::cout << "Key: " << it->first.displayKey() << endl;
564  if (debug_ > 5) {
565  std::cout << "Set: " << it->second.displaySet() << endl;
566  }
567  }
568  }
569  }

627  // }
628

631  if (debug_) {

645  ch->error() = 1;
646  }
647

653

659  } else if (Scheduler::instance().clock() -

669  }

672  ++it;
673  }

677 void
678 Uwal::startTx(Packet *p)
679 {
680  InterframeTmr.resched(interframe_period); // + (double)rand() /
681  // (double)RAND_MAX *
682  // interframe_period );
