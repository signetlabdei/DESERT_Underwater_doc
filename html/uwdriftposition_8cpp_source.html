117 void
118 UwDriftPosition::update(const double &now)
119 {
120  double t;
121 
122  for (t = nextUpdateTime_; t < now; t += updateTime_) {
123  // Calculate new speed
124  double vx_ = (alpha_ * old_speed_x_) +
125  (1 - alpha_) * (speed_horizontal_ +
126  deltax_ * (rand() / (static_cast<double>(
127  RAND_MAX) +
128  1.0)) *
129  getSign());
130  double vy_ = (alpha_ * old_speed_y_) +
131  (1 - alpha_) * (speed_longitudinal_ +
132  deltay_ * (rand() / (static_cast<double>(
133  RAND_MAX) +
134  1.0)) *
135  getSign());
136  double vz_ = (alpha_ * old_speed_z_) +
137  (1 - alpha_) * (speed_vertical_ +
138  deltaz_ * (rand() / (static_cast<double>(
139  RAND_MAX) +
140  1.0)) *
141  getSign());
142 
143  // Save the new speed in a variable
144  old_speed_x_ = vx_;
145  old_speed_y_ = vy_;
146  old_speed_z_ = vz_;
147 
148  // Calculate new position
149  double newx_ = x_ + (vx_ * updateTime_);
150  // cout << "-------->" << y_ << " + " << "(" << vy_ << " * " <<
151  // updateTime_ << ")";
152  double newy_ = y_ + (vy_ * updateTime_);
153  // cout << " = " << newy_ << endl;
154  double newz_ = z_ + (vz_ * updateTime_);
155 
156  // verify whether the new position has to be re-computed in order
157  // to maintain node position within the simulation field
158 
159  // Bounds check
160  if (boundx_ == 1) {
161  if (newx_ > xFieldWidth_) {
162  // newx_ = x_ + (- vx_ * updateTime_);
163  newx_ = (2 * xFieldWidth_) - newx_;
164  old_speed_x_ = (-old_speed_x_); // The sign of the current speed
165  // is inverted = Rebounce
166  speed_horizontal_ = (-speed_horizontal_); // The sign of the
167  // horizontal speed is
168  // inverted = Rebounce
169  // behaviour
170  } else if (newx_ < 0) {
171  // newx_ = x_ + (- vx_ * updateTime_);
172  newx_ = -newx_;
173  old_speed_x_ = (-old_speed_x_);
174  speed_horizontal_ = (-speed_horizontal_);
175  }
176  }
177  if (boundy_ == 1) {
178  if (newy_ > yFieldWidth_) {
179  // cout << "newy_ = " << newy_ << ". yFieldWidth_ = " <<
180  // yFieldWidth_ << endl;
181  // newy_ = y_ + (- vy_ * updateTime_);
182  newy_ = (2 * yFieldWidth_) - newy_;
183  old_speed_y_ = (-old_speed_y_);
184  speed_longitudinal_ = (-speed_longitudinal_);
185  // cout << "newy_ = " << newy_ << ". yFieldWidth_ = " <<
186  // yFieldWidth_ << endl;
187  } else if (newy_ < 0) {
188  // newy_ = y_ + (- vy_ * updateTime_);
189  newy_ = -newy_;
190  old_speed_y_ = (-old_speed_y_);
191  speed_longitudinal_ = (-speed_longitudinal_);
192  }
193  }
194  if (boundz_ == 1) {
195  if (newz_ < (-zFieldWidth_)) {
196  // cout << "newz_ = " << newz_ << ". zFieldWidth_ = " << -
197  // zFieldWidth_ << endl;
198  // newz_ = z_ + (- vz_ * updateTime_);
199  newz_ = (-zFieldWidth_) - (newz_ + (zFieldWidth_));
200  old_speed_z_ = -old_speed_z_;
201  speed_vertical_ = (-speed_vertical_);
202  // cout << "newz_ = " << newz_ << ". zFieldWidth_ = " << -
203  // zFieldWidth_ << endl;
204  } else if (newz_ > 0) {
205  // cout << "newz_ = " << newz_ << ". zFieldWidth_ = " << -
206  // zFieldWidth_ << endl;
207  // newz_ = z_ + (- vz_ * updateTime_);
208  newz_ = -newz_;
209  old_speed_z_ = -old_speed_z_;
210  speed_vertical_ = (-speed_vertical_);
211  // cout << "newz_ = " << newz_ << ". zFieldWidth_ = " << -
212  // zFieldWidth_ << endl;
213  }
214  }
215  if (debug_ > 10) {
216  printf("X:%.3f->%.3f Y:%.3f->%.3f Z:%.3f->%.3f\n",
217  x_,
218  newx_,
219  y_,
220  newy_,
221  z_,
222  newz_);
223  }
224  x_ = newx_;
225  y_ = newy_;
226  z_ = newz_;
227  }
228  nextUpdateTime_ = t;
229  if (debug_ > 10)
230  printf("nextUpdateTime = %f, now %f, updateTime %f\n",
231  nextUpdateTime_,
232  now,
233  updateTime_);
234 }

238 {
239  double rand_sign = (rand() / (static_cast<double>(RAND_MAX) + 1.0));
240  if (rand_sign < 0.5) {
241  return 1;
242  } else if (rand_sign > 0.5) {
243  return -1;
244  } else { // rand_sign == 0.5
245  return getSign();
