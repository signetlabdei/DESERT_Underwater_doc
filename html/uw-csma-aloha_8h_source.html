163 class AlohaTimer : public TimerHandler
164 {
165 public:
170 AlohaTimer(CsmaAloha *m)
171 : TimerHandler()
172 , start_time(0.0)
173 , left_duration(0.0)
174 , counter(0)
175 , module(m)
176 , timer_status(CSMA_IDLE)
177 {
178 assert(m != NULL);
179 }
180
184 virtual ~AlohaTimer()
185 {
186 }
187
191 virtual void
192 freeze()
193 {
194 assert(timer_status == CSMA_RUNNING);
195 left_duration -= (NOW - start_time);
196 if (left_duration <= 0.0)
197 left_duration = module->mac2phy_delay_;
198 force_cancel();
199 timer_status = CSMA_FROZEN;
200 }
201
206 virtual void
207 unFreeze()
208 {
209 assert(timer_status == CSMA_FROZEN);
210 start_time = NOW;
211 assert(left_duration > 0);
212 sched(left_duration);
213 timer_status = CSMA_RUNNING;
214 }
215
219 virtual void
220 stop()
221 {
222 timer_status = CSMA_IDLE;
223 force_cancel();
224 }
225
230 virtual void
231 schedule(double val)
232 {
233 start_time = NOW;
234 left_duration = val;
235 timer_status = CSMA_RUNNING;
236 resched(val);
237 }
238
243 bool
244 isIdle()
245 {
246 return (timer_status == CSMA_IDLE);
247 }
248
253 bool
254 isRunning()
255 {
256 return (timer_status == CSMA_RUNNING);
257 }
258
264 bool
265 isExpired()
266 {
267 return (timer_status == CSMA_EXPIRED);
268 }
269
274 bool
275 isFrozen()
276 {
277 return (timer_status == CSMA_FROZEN);
278 }
279
285 bool
286 isActive()
287 {
288 return (timer_status == CSMA_FROZEN ||
289 timer_status == CSMA_RUNNING);
290 }
291
296 void
297 resetCounter()
298 {
299 counter = 0;
300 }
301
306 void
307 incrCounter()
308 {
309 ++counter;
310 }
311
317 int
318 getCounter()
319 {
320 return counter;
321 }
322
327 double
328 getDuration()
329 {
330 return left_duration;
331 }
332
333 protected:
334 double start_time;
336 double left_duration;
338 int counter;
340 CsmaAloha *module;
341 CSMA_TIMER_STATUS timer_status;
342 };
