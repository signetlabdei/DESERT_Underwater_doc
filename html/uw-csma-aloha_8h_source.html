152 class AlohaTimer : public TimerHandler
153 {
154 public:
159 AlohaTimer(CsmaAloha *m)
160 : TimerHandler()
161 , start_time(0.0)
162 , left_duration(0.0)
163 , counter(0)
164 , module(m)
165 , timer_status(CSMA_IDLE)
166 {
167 assert(m != NULL);
168 }
169
173 virtual ~AlohaTimer() = default;
174
178 virtual void
179 freeze()
180 {
181 assert(timer_status == CSMA_RUNNING);
182 left_duration -= (NOW - start_time);
183 if (left_duration <= 0.0)
184 left_duration = module->mac2phy_delay_;
185 force_cancel();
186 timer_status = CSMA_FROZEN;
187 }
188
193 virtual void
194 unFreeze()
195 {
196 assert(timer_status == CSMA_FROZEN);
197 start_time = NOW;
198 assert(left_duration > 0);
199 sched(left_duration);
200 timer_status = CSMA_RUNNING;
201 }
202
206 virtual void
207 stop()
208 {
209 timer_status = CSMA_IDLE;
210 force_cancel();
211 }
212
217 virtual void
218 schedule(double val)
219 {
220 start_time = NOW;
221 left_duration = val;
222 timer_status = CSMA_RUNNING;
223 resched(val);
224 }
225
230 bool
231 isIdle()
232 {
233 return (timer_status == CSMA_IDLE);
234 }
235
240 bool
241 isRunning()
242 {
243 return (timer_status == CSMA_RUNNING);
244 }
245
251 bool
252 isExpired()
253 {
254 return (timer_status == CSMA_EXPIRED);
255 }
256
261 bool
262 isFrozen()
263 {
264 return (timer_status == CSMA_FROZEN);
265 }
266
271 bool
272 isActive()
273 {
274 return (timer_status == CSMA_FROZEN ||
275 timer_status == CSMA_RUNNING);
276 }
277
281 void
282 resetCounter()
283 {
284 counter = 0;
285 }
286
290 void
291 incrCounter()
292 {
293 ++counter;
294 }
295
300 int
301 getCounter() const
302 {
303 return counter;
304 }
305
310 double
311 getDuration() const
312 {
313 return left_duration;
314 }
315
316 protected:
317 double start_time;
319 double left_duration;
321 int counter;
323 CsmaAloha *module;
324 CSMA_TIMER_STATUS timer_status;
325 };
