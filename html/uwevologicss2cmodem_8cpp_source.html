116void
117UwEvoLogicsS2CModem::recv(Packet *p)
118{
119 // push new incoming pck into tx_queue
120 hdr_cmn *ch = HDR_CMN(p);
121 hdr_MPhy *ph = HDR_MPHY(p);
122 if (ch->direction() == hdr_cmn::UP) {
123 if (isOn == true) {
124 startRx(p);
125 endRx(p);
126 } else {
127 Packet::free(p);
128 }
129 } else { // DOWN
130 if (!isOn) {
131 return;
132 }
133 ph->Pr = 0;
134 ph->Pn = 0;
135 ph->Pi = 0;
136 ph->txtime = NOW;
137 ph->rxtime = ph->txtime;
138
139 ph->worth_tracing = false;
140
141 ph->srcSpectralMask = getTxSpectralMask(p);
142 ph->srcAntenna = getTxAntenna(p);
143 ph->srcPosition = getPosition();
144 ph->dstSpectralMask = 0;
145 ph->dstPosition = 0;
146 ph->dstAntenna = 0;
147 ph->modulationType = getModulationType(p);
148 ph->duration = getTxDuration(p);
149
150 std::unique_lock<std::mutex> tx_lock(tx_queue_m);
151 tx_queue.push(p);
152 tx_lock.unlock();
153 printOnLog(LogLevel::DEBUG,
154 "EVOLOGICSS2CMODEM",
155 "recv::PUSHING_IN_TX_QUEUE");
156 tx_queue_cv.notify_one();
157 }

162{
163 // Tcl &tcl = Tcl::instance();
164
165 if (argc == 2) {
166 if (!strcmp(argv[1], "start")) {
167 start();
168 return TCL_OK;
169 }
170 if (!strcmp(argv[1], "stop")) {
171 stop();
172 return TCL_OK;
173 }
174 if (!strcmp(argv[1], "setBurstMode")) {
175 tx_mode = TransmissionMode::BURST;
176 return TCL_OK;
177 }
178 if (!strcmp(argv[1], "setIMMode")) {
179 tx_mode = TransmissionMode::IM;
180 return TCL_OK;
181 }
182 if (!strcmp(argv[1], "enableIMAck")) {
183 ack_mode = true;
184 return TCL_OK;
185 }
186 if (!strcmp(argv[1], "disableIMAck")) {
187 ack_mode = false;
188 return TCL_OK;
189 }
190 if (!strcmp(argv[1], "initLUT")) {
191 initializeLUT();
192 return TCL_OK;
193 }
194 } else if (argc == 3) {
195 if (!strcmp(argv[1], "setSourceLevel")) {
196 std::stringstream sl_ss(argv[2]);
197 int sl_;
198 if (sl_ss >> sl_)
199 return TCL_OK;
200 else
201 return TCL_ERROR;
202 }
203 if (!strcmp(argv[1], "setTXDurationFileName")) {
204 std::string tmp_ = ((char *) argv[2]);
205 if (tmp_.size() == 0) {
206 fprintf(stderr, "Empty string for the file name");
207 return TCL_ERROR;
208 }
209 txdur_file_name_ = tmp_;
210 return TCL_OK;
211 }
212 if (!strcmp(argv[1], "setLUTSeparator")) {
213 string tmp_ = ((char *) argv[2]);
214 if (tmp_.size() == 0) {
215 fprintf(stderr, "Empty char for the file separator");
216 return TCL_ERROR;
217 }
218 txdur_token_separator_ = tmp_.at(0);
219 return TCL_OK;
220 }
221 }
222
223 return UwModem::command(argc, argv);

226void
227UwEvoLogicsS2CModem::initializeLUT()
228{
229 ifstream input_file_;
230 string line_;
231
232 input_file_.open(txdur_file_name_.c_str());
233
234 if (input_file_.is_open()) {
235 while (std::getline(input_file_, line_)) {
236 if (line_[0] == '#')
237 continue;
238
239 std::istringstream line_stream(line_);
240 std::string token;
241
242 std::getline(line_stream, token, txdur_token_separator_);
243 int size = std::atoi(token.c_str());
244 if (size < 1) {
245 printOnLog(LogLevel::ERROR,
246 "EVOLOGICSS2CMODEM",
247 "initializeLUT::INVALID_PKT_SIZE " + token);
248 continue;
249 }
250
251 std::getline(line_stream, token, txdur_token_separator_);
252 double duration = 0;
253 char *end {};
254 if (token.length() > 0) {
255 duration = std::strtod(token.c_str(), &end);
256
257 if (!(duration > 0))
258 printOnLog(LogLevel::ERROR,
259 "EVOLOGICSS2CMODEM",
260 "initializeLUT::INVALID_DURATION_FOR_PKT_SIZE "
261 + std::to_string(size));
262 }
263
264 std::getline(line_stream, token, txdur_token_separator_);
265 double proc_delay = 0;
266 if (token.length() > 0) {
267 proc_delay = std::strtod(token.c_str(), &end);
268 duration += proc_delay;
269
270 if (!(proc_delay > 0))
271 printOnLog(LogLevel::ERROR,
272 "EVOLOGICSS2CMODEM",
273 "initializeLUT::INVALID_PROC_DELAY_FOR_PKT_SIZE "
274 + std::to_string(size));
275 }
276
277 size2dur_[size] = duration;
278 }
279
280 initLUT_ = true;
281 } else {
282 printOnLog(LogLevel::ERROR,
283 "EVOLOGICSS2CMODEM",
284 "initializeLUT::FAIL_TO_READ_FROM_FILE " + txdur_file_name_);
285 }
286}

288int
289UwEvoLogicsS2CModem::recvSyncClMsg(ClMessage *m)
290{
291 if (m->type() == CLMSG_S2C_POWER_LEVEL) {
292 ClMsgS2CPowerLevel *msg = static_cast<ClMsgS2CPowerLevel *>(m);
293 if (msg->getReqType() == ClMsgUwPhy::SET_REQ) {
294 printOnLog(LogLevel::INFO,
295 "UWEVOLOGICSS2CMODEM",
296 "RECEIVED_CLMSG_FOR_POWER_CHANGE");
297 pend_source_level = msg->get_power_level();
298 source_level_change = true;
299 msg->setReqType(ClMsgUwPhy::SET_REPLY);
300 return 0;
301 } else if (msg->getReqType() == ClMsgUwPhy::GET_REQ) {
302 msg->set_power_level(curr_source_level);
303 msg->setReqType(ClMsgUwPhy::GET_REPLY);
304 return 0;
305 }
306 } else if (m->type() == CLMSG_S2C_TX_MODE) {
307 ClMsgS2CTxMode *msg = static_cast<ClMsgS2CTxMode *>(m);
308 if (msg->getReqType() == ClMsgUwPhy::SET_REQ) {
309 if (msg->get_tx_mode() == ClMsgS2CTxMode::S2C_TX_MODE_IM) {
310 tx_mode = TransmissionMode::BURST;
311 } else {
312 tx_mode = TransmissionMode::IM;
313 }
314 msg->setReqType(ClMsgUwPhy::SET_REPLY);
315 return 0;
316 } else if (msg->getReqType() == ClMsgUwPhy::GET_REQ) {
317 if (tx_mode == TransmissionMode::IM) {
318 msg->set_tx_mode(ClMsgS2CTxMode::S2C_TX_MODE_IM);
319 } else {
320 msg->set_tx_mode(ClMsgS2CTxMode::S2C_TX_MODE_BURST);
321 }
322 msg->setReqType(ClMsgUwPhy::GET_REPLY);
323 return 0;
324 }
325 } else if (m->type() == CLMSG_S2C_RX_FAILED) {
326 ClMsgS2CRxFailed *msg = static_cast<ClMsgS2CRxFailed *>(m);
327 if (msg->getReqType() == ClMsgUwPhy::GET_REQ) {
328 msg->set_n_rx_failed(n_rx_failed);
329 msg->setReqType(ClMsgUwPhy::GET_REPLY);
330 return 0;
331 } else {
332 return 1;
333 }
334 } else if (m->type() == CLMSG_MAC2PHY_GETTXDURATION) {
335 Packet *p = ((ClMsgMac2PhyGetTxDuration *) m)->pkt;
336 hdr_cmn *ch = HDR_CMN(p);
337
338 if (ch->direction() == hdr_cmn::DOWN) {
339 double duration = getTxDuration(p);
340
341 if (duration > 0)
342 printOnLog(LogLevel::INFO,
343 "EVOLOGICSS2CMODEM",
344 "recvSyncClMsg::SET_TXDURATION " +
345 std::to_string(duration));
346
347 ((ClMsgMac2PhyGetTxDuration *) m)->setDuration(duration);
348
349 return 0;
350 }
351
352 // Don't set TX duration on RX.
353 ((ClMsgMac2PhyGetTxDuration *) m)->setDuration(-1);
354
355 return 1;
356 }
357
358 return MPhy::recvSyncClMsg(m);
359}

377bool
378UwEvoLogicsS2CModem::configure(Config cmd)
379{
380 std::string config_cmd{""};
381
382 switch (cmd) {
383 case Config::ATL_GET: {
384 std::string command = p_interpreter->buildGetATL();
385 break;
386 }
387 case Config::ATL_SET: {
388 std::string command = p_interpreter->buildSetATL(pend_source_level);
389 break;
390 }
391 case Config::ATAL_GET: {
392 std::string command = p_interpreter->buildGetATAL();
393 break;
394 }
395 case Config::ATAL_SET: {
396 std::string command = p_interpreter->buildSetATAL(modemID);
397 break;
398 }
399 case Config::MODEM_STATUS: {
400 std::string command = p_interpreter->buildATS();
401 break;
402 }
403 case Config::CURR_SETTINGS: {
404 std::string command = p_interpreter->buildATV();
405 break;
406 }
407 default:
408 return false;
409 }
410
411 std::unique_lock<std::mutex> state_lock(status_m);
412
413 if (status_cv.wait_for(state_lock, MODEM_TIMEOUT, [&] {
414 return status == ModemState::AVAILABLE;
415 })) {
416
417 status = ModemState::BUSY;
418
419 std::lock_guard<std::mutex> tx_state_lock(tx_status_m);
420
421 if (!p_connector->writeToDevice(config_cmd)) {
422 printOnLog(LogLevel::ERROR,
423 "EVOLOGICSS2CMODEM",
424 "configure::FAIL_TO_WRITE_TO_DEVICE=" + config_cmd);
425 return false;
426 }
427
428 state_lock.unlock();
429 tx_status = TransmissionState::TX_PENDING;
430
431 } else {
432
433 return false;
434 }
435
436 return true;
437}

439void
440UwEvoLogicsS2CModem::startTx(Packet *p)
441{
442 // this method does nothing in ns, so it can be called from an
443 // external thread
444 // save MAC and AL headers and write payload
445 hdr_mac *mach = HDR_MAC(p);
446 hdr_uwal *uwalh = HDR_UWAL(p);
447 std::string payload;
448 payload.assign(uwalh->binPkt(), uwalh->binPktLength());
449
450 // build command to perform a SEND or SENDIM
451 std::string cmd_s;
452 if (tx_mode == TransmissionMode::IM) {
453 cmd_s = p_interpreter->buildSendIM(payload, mach->macDA(), ack_mode);
454 } else {
455 cmd_s = p_interpreter->buildSend(payload, mach->macDA());
456 }
457
458 printOnLog(LogLevel::INFO,
459 "EVOLOGICSS2CMODEM",
460 "startTx::COMMAND_TX::" + cmd_s);
461
462 // write the obtained command to the device, through the connector
463 std::unique_lock<std::mutex> state_lock(status_m);
464 if (status_cv.wait_for(state_lock, MODEM_TIMEOUT, [&] {
465 return status == ModemState::AVAILABLE;
466 })) {
467
468 status = ModemState::BUSY;
469
470 std::unique_lock<std::mutex> tx_state_lock(tx_status_m);
471
472 if ((p_connector->writeToDevice(cmd_s)) < 0) {
473 printOnLog(LogLevel::ERROR,
474 "EVOLOGICSS2CMODEM",
475 "startTx::FAIL_TO_WRITE_TO_DEVICE=" + cmd_s);
476 return;
477 }
478
479 tx_status = TransmissionState::TX_PENDING;
480 status_cv.wait_for(state_lock, MODEM_TIMEOUT, [&] {
481 return status == ModemState::AVAILABLE;
482 });
483 state_lock.unlock();
484
485 if (tx_mode == TransmissionMode::IM) {
486
487 size_t status_polling_counter = 0;
488 im_status_updated.store(false);
489 while (status_polling_counter < MAX_N_STATUS_QUERIES &&
490 tx_status != TransmissionState::TX_IDLE) {
491
492 cmd_s = p_interpreter->buildATDI();
493
494 printOnLog(LogLevel::INFO,
495 "EVOLOGICSS2CMODEM",
496 "startTx::SENDING=" + cmd_s);
497
498 if ((p_connector->writeToDevice(cmd_s)) < 0) {
499 printOnLog(LogLevel::ERROR,
500 "EVOLOGICSS2CMODEM",
501 "startTx::FAIL_TO_WRITE_TO_DEVICE=" + cmd_s);
502 }
503
504 tx_status_cv.wait_for(tx_state_lock, WAIT_DELIVERY_IM, [&] {
505 return (tx_status == TransmissionState::TX_IDLE);
506 });
507
508 if (tx_status == TransmissionState::TX_IDLE) {
509 printOnLog(LogLevel::DEBUG,
510 "EVOLOGICSS2CMODEM",
511 "startTx::TX_IDLE");
512 break;
513 } else {
514 im_status_updated.store(false);
515 printOnLog(LogLevel::DEBUG,
516 "EVOLOGICSS2CMODEM",
517 "startTx::TX_PENDING");
518 }
519
520 status_polling_counter++;
521 }
522
523 // In case the attempts reached MAX_N_STATUS_QUERIES, force
524 // tx_status
525 if (status_polling_counter == MAX_N_STATUS_QUERIES &&
526 tx_status != TransmissionState::TX_IDLE) {
527 printOnLog(LogLevel::ERROR,
528 "EVOLOGICSS2CMODEM",
529 "startTx::MAX_N_STATUS_QUERIES_REACHED");
530 }
531
532 } else { // tx_mode == TransmissionMode::BURST
533 tx_status_cv.wait_for(tx_state_lock, WAIT_DELIVERY_BURST, [&] {
534 return (tx_status == TransmissionState::TX_IDLE);
535 });
536 }
537
538 std::function<void(UwModem &, Packet * p)> callback =
539 &UwModem::realTxEnded;
540 ModemEvent e = {callback, p};
541 event_q.push(e);
542
543 } else {
544 printOnLog(LogLevel::ERROR,
545 "EVOLOGICSS2CMODEM",
546 "startTx::TIMEOUT_EXPIRED::FORCING_MODEM_AVAILABILITY");
547 status = ModemState::AVAILABLE;
548 }
549}

567void
568UwEvoLogicsS2CModem::start()
569{
570 if (modem_address == "") {
571 std::cout << "ERROR: Modem address not set!" << std::endl;
572 printOnLog(
573 LogLevel::ERROR, "EVOLOGICSS2CMODEM", "start::ADDRESS_NOT_SET");
574 return;
575 }
576
577 if (!p_connector->openConnection(modem_address)) {
578 std::cout << "ERROR: connection to modem failed to open: "
579 << modem_address << std::endl;
580 printOnLog(LogLevel::ERROR,
581 "EVOLOGICSS2CMODEM",
582 "start::CONNECTION_OPEN_FAILED");
583 return;
584 }
585
586 // set flags to true so loops can start
587 receiving.store(true);
588 transmitting.store(true);
589
590 // branch off threads
591 rx_thread = std::thread(&UwEvoLogicsS2CModem::receivingData, this);
592
593 tx_thread = std::thread(&UwEvoLogicsS2CModem::transmittingData, this);
594
595 checkTimer = new CheckTimer(this);
596 checkTimer->resched(period);
597}

602
603 std::cout << "UwEvoLogicsS2CModem::stop() close conn " << std::endl;

612 printOnLog(LogLevel::ERROR,
613 "EVOLOGICSS2CMODEM",

616 if (rx_thread.joinable()) {
617 rx_thread.join();
618 }
619
620 checkTimer->force_cancel();
621}

623void
624UwEvoLogicsS2CModem::transmittingData()
625{
626 while (transmitting.load()) {
627
628 std::unique_lock<std::mutex> tx_lock(tx_queue_m);
629 tx_queue_cv.wait(
630 tx_lock, [&] { return !tx_queue.empty() || !transmitting; });
631 if (!transmitting) {
632 break;
633 }
634
635 Packet *pck = tx_queue.front();
636 tx_queue.pop();
637 tx_lock.unlock();
638 if (pck) {
639 startTx(pck);
640 }
641
642 printOnLog(LogLevel::DEBUG,
643 "EVOLOGICSS2CMODEM",
644 "transmittingData::BLOCKING_ON_NEXT_PACKET");
645 }
646}

648void
649UwEvoLogicsS2CModem::receivingData()
650{
651 data_buffer.resize(DATA_BUFFER_LEN);
652 std::fill(data_buffer.begin(), data_buffer.end(), '\0');
653 // iterators that keep track of read/write operations
654 std::vector<char>::iterator beg_it = data_buffer.begin();
655 std::vector<char>::iterator end_it = data_buffer.begin();
656 // iterators that keep track of commands research
657 std::vector<char>::iterator cmd_b = data_buffer.begin();
658 std::vector<char>::iterator cmd_e = data_buffer.begin();
659 UwInterpreterS2C::Response cmd = UwInterpreterS2C::Response::NO_COMMAND;
660 int r_bytes = 0;
661 int offset = 0;
662
663 while (receiving.load()) {
664
665 r_bytes = p_connector->readFromDevice(
666 &(*beg_it) + offset, MAX_READ_BYTES - r_bytes);
667
668 end_it = beg_it + r_bytes + offset;
669
670 while (receiving.load() && r_bytes > 0 &&
671 (cmd = p_interpreter->findResponse(beg_it, end_it, cmd_b)) !=
672 UwInterpreterS2C::Response::NO_COMMAND) {
673
674 while (receiving.load() &&
675 !p_interpreter->parseResponse(
676 cmd, end_it, cmd_b, cmd_e, rx_payload)) {
677
678 int new_r_bytes = p_connector->readFromDevice(
679 &(*end_it), MAX_READ_BYTES - r_bytes);
680
681 r_bytes += new_r_bytes;
682 end_it = beg_it + r_bytes;
683 }
684
685 printOnLog(LogLevel::DEBUG,
686 "EVOLOGICSS2CMODEM",
687 "receivingData::RX_MSG=" + std::string(cmd_b, cmd_e));
688
689 updateStatus(cmd);
690 std::fill(cmd_b, cmd_e, '\0');
691 cmd_b = cmd_e + 1;
692 }
693
694 // check if bytes are left after parsing and move them to beginning
695 if ((int) (cmd_e - beg_it) < r_bytes) {
696 offset = r_bytes - (int) (cmd_e - beg_it);
697 std::copy(cmd_e, cmd_e + offset, data_buffer.begin());
698 } else {
699 offset = 0;
700 }
701 }
702}

704void
705UwEvoLogicsS2CModem::updateStatus(UwInterpreterS2C::Response cmd)
706{
707 std::unique_lock<std::mutex> state_lock(status_m);
708 switch (cmd) {
709
710 case UwInterpreterS2C::Response::RECVIM: {
711
712 status = ModemState::AVAILABLE;
713 status_cv.notify_all();
714 state_lock.unlock();
715
716 Packet *p = Packet::alloc();
717 createRxPacket(p);
718 std::function<void(UwModem &, Packet * p)> callback =
