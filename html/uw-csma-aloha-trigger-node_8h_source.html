170  virtual void
171  freeze()
172  {
173  assert(timer_status == UW_CS_ALOHA_TRIG_NODE_RUNNING);
174  left_duration -= (NOW - start_time);
175  if (left_duration <= 0.0)
176  left_duration = module->mac2phy_delay_;
177  force_cancel();
178  timer_status = UW_CS_ALOHA_TRIG_NODE_FROZEN;
179  }
180 
185  virtual void
186  unFreeze()
187  {
188  assert(timer_status == UW_CS_ALOHA_TRIG_NODE_FROZEN);
189  start_time = NOW;
190  assert(left_duration > 0);
191  sched(left_duration);
192  timer_status = UW_CS_ALOHA_TRIG_NODE_RUNNING;
193  }
194 
198  virtual void
199  stop()
200  {
201  timer_status = UW_CS_ALOHA_TRIG_NODE_IDLE;
202  force_cancel();
203  }
204 
209  virtual void
210  schedule(double val)
211  {
212  start_time = NOW;
213  left_duration = val;
214  timer_status = UW_CS_ALOHA_TRIG_NODE_RUNNING;
215  resched(val);
216  }
