142 class Csma_Aloha_Triggered_Timer : public TimerHandler
143 {
144 public:
149 Csma_Aloha_Triggered_Timer(UwCsmaAloha_Trigger_NODE *m)
150 : TimerHandler()
151 , start_time(0.0)
152 , left_duration(0.0)
153 , counter(0)
154 , module(m)
155 , timer_status(UW_CS_ALOHA_TRIG_NODE_IDLE)
156 {
157 assert(m != NULL);
158 }
159
163 virtual ~Csma_Aloha_Triggered_Timer()
164 {
165 }
166
170 virtual void
171 freeze()
172 {
173 assert(timer_status == UW_CS_ALOHA_TRIG_NODE_RUNNING);
174 left_duration -= (NOW - start_time);
175 if (left_duration <= 0.0)
176 left_duration = module->mac2phy_delay_;
177 force_cancel();
178 timer_status = UW_CS_ALOHA_TRIG_NODE_FROZEN;
179 }
180
185 virtual void
186 unFreeze()
187 {
188 assert(timer_status == UW_CS_ALOHA_TRIG_NODE_FROZEN);
189 start_time = NOW;
190 assert(left_duration > 0);
191 sched(left_duration);
192 timer_status = UW_CS_ALOHA_TRIG_NODE_RUNNING;
193 }
194
198 virtual void
199 stop()
200 {
201 timer_status = UW_CS_ALOHA_TRIG_NODE_IDLE;
202 force_cancel();
203 }
204
209 virtual void
210 schedule(double val)
211 {
212 start_time = NOW;
213 left_duration = val;
214 timer_status = UW_CS_ALOHA_TRIG_NODE_RUNNING;
215 resched(val);
216 }
217
222 bool
223 isIdle()
224 {
225 return (timer_status == UW_CS_ALOHA_TRIG_NODE_IDLE);
226 }
227
232 bool
233 isRunning()
234 {
235 return (timer_status == UW_CS_ALOHA_TRIG_NODE_RUNNING);
236 }
237
242 bool
243 isExpired()
244 {
245 return (timer_status == UW_CS_ALOHA_TRIG_NODE_EXPIRED);
246 }
247
252 bool
253 isFrozen()
254 {
255 return (timer_status == UW_CS_ALOHA_TRIG_NODE_FROZEN);
256 }
257
262 bool
263 isActive()
264 {
265 return (timer_status == UW_CS_ALOHA_TRIG_NODE_FROZEN ||
266 timer_status == UW_CS_ALOHA_TRIG_NODE_RUNNING);
267 }
268
272 void
273 resetCounter()
274 {
275 counter = 0;
276 }
277
282 void
283 incrCounter()
284 {
285 ++counter;
286 }
287
292 int
293 getCounter()

295 return counter;
