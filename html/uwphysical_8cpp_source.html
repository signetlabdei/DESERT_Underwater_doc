208 UnderwaterPhysical::recv(Packet *p)
209 {
210  hdr_cmn *ch = HDR_CMN(p);
211  hdr_MPhy *ph = HDR_MPHY(p);
212 
213  if (ch->direction() == hdr_cmn::UP) {
214  ph->dstSpectralMask = getRxSpectralMask(p);
215  ph->dstPosition = getPosition();
216  ph->dstAntenna = getRxAntenna(p);
217 
218  assert(ph->dstSpectralMask);
219  assert(ph->dstPosition);
220 
221  ph->Pr = getRxPower(p);
222 
223  p->txinfo_.RxPr = 0;
224  p->txinfo_.CPThresh = 0;
225 
226  if (ph->Pr > 0) {
227  ph->Pn = getNoisePower(p);
228 
229  if (interference_) {
230  interference_->addToInterference(p);
231  }
232 
233  ph->rxtime = NOW;
234  ph->worth_tracing = true;
235 
236  if (isOn == true) {
237  PacketEvent *pe = new PacketEvent(p);
238  Scheduler::instance().schedule(&rxtimer, pe, ph->duration);
239 
240  startRx(p);
241  } else {
242  Packet::free(p);
243  }
244  } else {
245  Packet::free(p);
246  }
247  } else { // Direction DOWN
248  assert(isOn);
249 
250  ph->Pr = 0;
251  ph->Pn = 0;
252  ph->Pi = 0;
253  ph->txtime = NOW;
254  ph->rxtime = ph->txtime;
255 
256  ph->worth_tracing = false;
257 
258  ph->srcSpectralMask = getTxSpectralMask(p);
259  ph->srcAntenna = getTxAntenna(p);
260  ph->srcPosition = getPosition();
261  ph->dstSpectralMask = 0;
262  ph->dstPosition = 0;
263  ph->dstAntenna = 0;
264  ph->modulationType = getModulationType(p);
265  ph->duration = getTxDuration(p);
266 
267  ph->Pt = getTxPower(p);
268 
269  assert(ph->srcSpectralMask);
270  assert(ph->srcPosition);
271  assert(ph->duration > 0);
272  assert(ph->Pt > 0);
273 
274  PacketEvent *pe = new PacketEvent(p->copy());

295 void
296 UnderwaterPhysical::startRx(Packet *p)
297 {
298  hdr_mac *mach = HDR_MAC(p);
299  hdr_MPhy *ph = HDR_MPHY(p);
300 
301  static int mac_addr = -1;
302 
303  ClMsgPhy2MacAddr msg;
304  sendSyncClMsg(&msg);
305  mac_addr = msg.getAddr();
306 
307  if ((PktRx == 0) && (txPending == false)) {
308  // The receiver is is not synchronized on any transmission
309  // so we can sync on this packet
310  double snr_dB = 10 * log10(ph->Pr / ph->Pn);
311  if (debug_)
312  std::cout << NOW << " UnderwaterPhysical(" << mac_addr
313  << ")::startRx() "
314  << "snr_dB = " << snr_dB
315  << "; AcquisitionThreshold_dB_ = "
316  << getAcquisitionThreshold() << " pr "
317  << 10 * log10(ph->Pr) << " pn " << 10 * log10(ph->Pn)
318  << " end " << NOW + HDR_CMN(p)->txtime() << " src "
319  << HDR_CMN(p)->prev_hop_ << " dest "
320  << HDR_CMN(p)->next_hop() << " size "
321  << HDR_CMN(p)->size() << std::endl;
322  if (snr_dB > getAcquisitionThreshold()) {
323  if (ph->modulationType == modid) {
324  // This is a BPSK packet so we sync on it
325  PktRx = p;
326  // Notify the MAC
327  Phy2MacStartRx(p);
328  if (debug_)
329  std::cout << NOW << " UnderwaterPhysical(" << mac_addr
330  << ")::startRx() "
331  << " sync on PktRx = " << PktRx << " end "
332  << NOW + HDR_CMN(p)->txtime() << " src "
333  << HDR_CMN(p)->prev_hop_ << " dest "
334  << HDR_CMN(p)->next_hop() << " size "
335  << HDR_CMN(p)->size() << std::endl;
336  return;
337  } else {
338  if (debug_)
339  std::cout << NOW << " UnderwaterPhysical(" << mac_addr
340  << ")::startRx() dropping pkt, wrong mod id"
341  << std::endl;
342  if ((mach->macDA() == mac_addr) &&
343  (mach->ftype() != MF_CONTROL)) {
344  incrTot_pkts_lost();
345  } else if ((mach->macDA() == mac_addr) &&
346  (mach->ftype() == MF_CONTROL)) {
347  incrTotCrtl_pkts_lost();
348  }
349  }
350  } else {
351  if (debug_)
352  std::cout << NOW << " UnderwaterPhysical(" << mac_addr
353  << ")::startRx() dropping pkt, below threshold"
354  << std::endl;
355  incrErrorPktsNoise();
356  if (mach->ftype() != MF_CONTROL) {
357  incrTot_pkts_lost();
358  } else if (mach->ftype() == MF_CONTROL) {
359  incrTotCrtl_pkts_lost();
360  }
361  }
362  } else if (txPending == true) {
363  if (debug_)
364  std::cout << NOW << " UnderwaterPhysical(" << mac_addr
365  << ")::startRx() dropping pkt, tx pending" << std::endl;
366  if (mach->ftype() != MF_CONTROL) {
367  incrTot_pkts_lost();
368  }
369  if (mach->ftype() == MF_CONTROL) {
370  incrTotCrtl_pkts_lost();
371  }
372  } else {
373  if (mach->ftype() != MF_CONTROL) {
374  incrTot_pkts_lost();
375  } else if (mach->ftype() == MF_CONTROL) {

381 void
382 UnderwaterPhysical::endRx(Packet *p){
383  hdr_cmn *ch = HDR_CMN(p);
384  hdr_MPhy *ph = HDR_MPHY(p);
385  hdr_mac *mach = HDR_MAC(p);
386  counter interferent_pkts;
387 
388  static int mac_addr = -1;
389 
390  ClMsgPhy2MacAddr msg;
391  sendSyncClMsg(&msg);
392  mac_addr = msg.getAddr();
393 
394  if (PktRx != 0) {
395  if (PktRx == p) {
396  double per_ni;
397  int nbits = ch->size() * 8;
398  double x = RNG::defaultrng()->uniform_double();
399  bool error_n = 0; // x <= per_n;
400  bool error_ni = 0;
401  double interference_power = 0;
402  if (interference_) {
403  if (Interference_Model == "CHUNK") {
404  const PowerChunkList &power_chunk_list =
405  interference_->getInterferencePowerChunkList(p);
406  if (power_chunk_list.size() < 1) {
407  // we have no interferent
408  per_ni = getPER((ph->Pr / ph->Pn), nbits, p);
409  error_ni = x <= per_ni;
410  } else {
411  for (PowerChunkList::const_iterator itInterf =
412  power_chunk_list.begin();
413  itInterf != power_chunk_list.end();
414  itInterf++) {
415  int nbits2 = itInterf->second * BitRate_;
416  interference_power = itInterf->first;
417  per_ni = getPER(
418  ph->Pr / (ph->Pn + itInterf->first), nbits2, p);
419  x = RNG::defaultrng()->uniform_double();
420  error_ni = x <= per_ni;
421  if (error_ni) {
422  break;
423  }
424  }
425  }
426  } else if (Interference_Model == "MEANPOWER") {
427  interference_power = interference_->getInterferencePower(p);
428  per_ni = getPER(
429  ph->Pr / (ph->Pn + interference_power), nbits, p);
430  error_ni = x <= per_ni;
431  } else {
432  std::cerr << "Please choose the right interference model "
433  "to use: CHUNK or MEANPOWER"
434  << std::endl;
435  exit(1);
436  }
437  interferent_pkts = interference_->getCounters(p);
438 
439  } else {
440  interference_power = ph->Pi;
441  per_ni = getPER(ph->Pr / (ph->Pn + ph->Pi), nbits, p);
442  error_ni = x <= per_ni;
443  }
444 
445  if (interference_power < (ph->Pn / 10)) {
446  error_n = error_ni;
447  error_ni = 0; // error transfered on noise
448  }
449  //update stats and trigger the modules that collect the stats
450  dynamic_cast<UwPhysicalStats *>(stats_ptr)->updateStats(getId(),
451  getStackId(), ph->Pr, ph->Pn, interference_power,
452  (error_n>0||error_ni>0));
453  ClMsgTriggerStats m = ClMsgTriggerStats();
454  sendSyncClMsg(&m);
455 
456  if (time_ready_to_end_rx_ > Scheduler::instance().clock()) {
457  Rx_Time_ = Rx_Time_ + ph->duration - time_ready_to_end_rx_ +
458  Scheduler::instance().clock();
459  } else {
460  Rx_Time_ += ph->duration;
461  }
462  time_ready_to_end_rx_ =
463  Scheduler::instance().clock() + ph->duration;
464  Energy_Rx_ += consumedEnergyRx(ph->duration);
465  ch->error() = error_n || error_ni;
466  if (debug_) {
467  if (error_ni == 1) {
468  std::cout
469  << NOW << " UnderwaterPhysical(" << mac_addr
470  << ")::endRx() packet " << ch->uid()
471  << " contains errors due to noise and interference."
472  << std::endl;
473  } else if (error_n == 1) {
474  std::cout << NOW << " UnderwaterPhysical(" << mac_addr
475  << ")::endRx() packet " << ch->uid()
476  << " contains errors due to noise." << std::endl;
477  }
478  }
479  if (error_n) {
480  incrErrorPktsNoise();
481  if (mach->ftype() != MF_CONTROL) {
482  incrTot_pkts_lost();
483  } else if (mach->ftype() == MF_CONTROL) {
484  incrTotCrtl_pkts_lost();
485  }
486  } else if (error_ni) {
487  if (mach->ftype() != MF_CONTROL) {
488  incrErrorPktsInterf();
489  incrTot_pkts_lost();
490  if (interferent_pkts.second >= 1) {
491  incrCollisionDATA();
492  } else {
493  if (interferent_pkts.first > 0) {
494  incrCollisionDATAvsCTRL();
495  }
496  }
497  } else if (mach->ftype() == MF_CONTROL) {
498  incrTotCrtl_pkts_lost();
499  incrErrorCtrlPktsInterf();
500  if (interferent_pkts.first > 0) {
501  incrCollisionCTRL();
502  }
503  }
504  }
505  sendUp(p);
506  PktRx = 0;
507  } else {
508  dropPacket(p);
509  }
510  } else {
511  dropPacket(p);
512  }
513 } /* UnderwaterPhysical::endRx */

515 double
516 UnderwaterPhysical::getPER(double _snr, int _nbits, Packet *_p)
517 {
518  double snr_with_penalty = _snr * pow(10, RxSnrPenalty_dB_ / 10.0);
519 
520  double ber_ = 0;
521  if (modulation_name_ == "BPSK") {
522  ber_ = 0.5 * erfc(sqrt(snr_with_penalty));
523  } else if (modulation_name_ == "BFSK") {
524  ber_ = 0.5 * exp(-snr_with_penalty / 2);
525  } else if (modulation_name_ == "8PSK") {
526  double const M = 8;
527  ber_ = (1 / this->log2(M)) *
528  get_prob_error_symbol_mpsk(snr_with_penalty, M);
529  } else if (modulation_name_ == "16PSK") {
530  double const M = 16;
531  ber_ = (1 / this->log2(M)) *
532  get_prob_error_symbol_mpsk(snr_with_penalty, M);
533  } else if (modulation_name_ == "32PSK") {
534  double const M = 32;
535  ber_ = (1 / this->log2(M)) *
536  get_prob_error_symbol_mpsk(snr_with_penalty, M);
537  }
538 
539  // PER calculation
540  return 1 - pow(1 - ber_, _nbits);
541 } /* UnderwaterPhysical::getPER */
