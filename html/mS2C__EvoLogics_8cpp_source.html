129 modem_state_t
130 MS2C_EvoLogics::check_modem()
131 {
132 
133  modem_state_t modemStatus_old = pmDriver->getStatus();
134  modem_state_t modemStatus = pmDriver->updateStatus();
135  std::stringstream sstr("");
136  string strlog;
137  sstr << "CHECK_MODEM::TRANSITION_FROM_" << modemStatus_old << "_TO_"
138  << modemStatus;
139  sstr >> strlog;
140  pmDriver->printOnLog(LOG_LEVEL_INFO, "UWMPHY_MODEM", strlog);
141 
142  if (modemStatus == MODEM_CHANGE_POWER_LEVEL) {
143  if (modemStatus_old == MODEM_TX) {
144  endTx(popTxBuff());
145  }
146  static_cast<MdriverS2C_EvoLogics *>(pmDriver)->change_power_level();
147  return modemStatus;
148  }
149 
150  if (modemStatus == MODEM_RX && modemStatus_old == MODEM_IDLE) {
151  startRx(PktRx);
152  return modemStatus;
153  }
154 
155  if (modemStatus == MODEM_IDLE_RX && modemStatus_old == MODEM_RX) {
156  if (modemStatus_old == MODEM_RX) {
157  endRx(PktRx);
158  return modemStatus;
159  }
160  if (modemStatus_old == MODEM_TX_RX) {
161  endTx(popTxBuff());
162  startRx(PktRx);
163  endRx(PktRx);
164  return modemStatus;
165  }
166  }
167 
168  if (modemStatus == MODEM_CFG && modemStatus_old == MODEM_CFG) {
169  pmDriver->modemSetID();
170  return modemStatus;
171  }
172 
173  if (modemStatus == MODEM_NOISE) {
174  if (modemStatus_old == MODEM_NOISE) {
175  static_cast<MdriverS2C_EvoLogics *>(pmDriver)->probeNoise();
176  return modemStatus;
177  } else {
178  if (modemStatus_old == MODEM_TX) {
179  endTx(popTxBuff());
180  }
181  static_cast<MdriverS2C_EvoLogics *>(pmDriver)->enterNoiseState();
182  return modemStatus;
183  }
184  }
185 
186  if (modemStatus == MODEM_IDLE) {
187  if (modemStatus_old == MODEM_NOISE) {
188  static_cast<MdriverS2C_EvoLogics *>(pmDriver)->enterListenState();
189  return modemStatus;
190  }
191  if (modemStatus_old == MODEM_TX) {
192  endTx(popTxBuff());
193  return modemStatus;
194  }
195  if (modemStatus_old == MODEM_MULTIPATH ||
196  modemStatus_old == MODEM_IDLE ||
197  modemStatus_old == MODEM_CHANGE_POWER_LEVEL) {
198  return modemStatus;
199  }
200  }
201 
202  if (modemStatus == MODEM_MULTIPATH) {
203  if (modemStatus_old == MODEM_TX) {
204  endTx(popTxBuff());
205  }
206  static_cast<MdriverS2C_EvoLogics *>(pmDriver)->probeMultipath();
207  return modemStatus;
208  }
209 
210  pmDriver->printOnLog(
211  LOG_LEVEL_INFO, "UWMPHY_MODEM", "STATE_CHANGE_NOT_HANDLED");
212  return modemStatus;
213 }

215 int
216 MS2C_EvoLogics::recvSyncClMsg(ClMessage *m)
217 {
218  if (m->type() == CLMSG_S2C_POWER_LEVEL) {
219  ClMsgS2CPowerLevel *msg = static_cast<ClMsgS2CPowerLevel *>(m);
220  if (msg->getReqType() == ClMsgUwPhy::SET_REQ) {
221  pmDriver->printOnLog(LOG_LEVEL_INFO,
222  "UWPHY_MODEM",
223  "RECEIVED_CLMSG_FOR_POWER_CHANGE");
224  static_cast<MdriverS2C_EvoLogics *>(pmDriver)
225  ->power_level_change_ind(msg->get_power_level());
226  msg->setReqType(ClMsgUwPhy::SET_REPLY);
227  return (0);
228  } else if (msg->getReqType() == ClMsgUwPhy::GET_REQ) {
229  int pwl;
230  pwl = static_cast<MdriverS2C_EvoLogics *>(pmDriver)
231  ->get_actual_power_level();
232  msg->set_power_level(pwl);
233  msg->setReqType(ClMsgUwPhy::GET_REPLY);
234  return (0);
235  }
236  } else if (m->type() == CLMSG_S2C_TX_MODE) {
237  ClMsgS2CTxMode *msg = static_cast<ClMsgS2CTxMode *>(m);
238  if (msg->getReqType() == ClMsgUwPhy::SET_REQ) {
239  /*TODO: Method to handle change of transmission method */
240  msg->setReqType(ClMsgUwPhy::SET_REPLY);
241  return (0);
242  } else if (msg->getReqType() == ClMsgUwPhy::GET_REQ) {
243  /*TODO: Global variable to store current tx type */
244  msg->set_tx_mode(ClMsgS2CTxMode::S2C_TX_MODE_IM);
245  msg->setReqType(ClMsgUwPhy::GET_REPLY);
246  return (0);
247  }
248  }
249 
250  return MPhy::recvSyncClMsg(m);
251 }
