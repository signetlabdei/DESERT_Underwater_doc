210void
211UwCsmaAloha_Trigger_NODE::recvFromUpperLayers(Packet *p)
212{
213 if ((has_buffer_queue == true) && ((Q.size()) < (buffer_pkts)) ||
214 (has_buffer_queue == false)) {
215 initPkt(p);
216 Q.push(p);
217 incrUpperDataRx();
218 waitStartTime();
219
220 if (curr_state == UW_CS_ALOHA_TRIG_NODE_STATE_IDLE) {
221 refreshReason(UW_CS_ALOHA_TRIG_NODE_REASON_DATA_PENDING);
222 stateListen();
223 } else {
224 if (debug_)
225 cerr << showpoint << NOW << " " << __PRETTY_FUNCTION__
226 << " transmitting packet" << endl;
227 }
228 } else {
229 incrDiscardedPktsTx();
230 drop(p, 1, UW_CS_ALOHA_TRIG_NODE_DROP_REASON_BUFFER_FULL);
231 }

313void
314UwCsmaAloha_Trigger_NODE::Phy2MacEndRx(Packet *p)
315{
316 hdr_cmn *ch = HDR_CMN(p);
317 packet_t rx_pkt_type = ch->ptype();
318 hdr_mac *mach = HDR_MAC(p);
319 hdr_MPhy *ph = HDR_MPHY(p);
320
321 int dest_mac = mach->macDA();
322
323 double gen_time = ph->txtime;
324 double received_time = ph->rxtime;
325 double diff_time = received_time - gen_time;
326
327 double distance = diff_time * prop_speed;
328
329 if (debug_)
330 cout << NOW << " UwCsmaAloha_Trigger_NODE(" << addr
331 << ")::Phy2MacEndRx() " << status_info[curr_state]
332 << ", received a pkt type = " << ch->ptype()
333 << ", src addr = " << mach->macSA()
334 << " dest addr = " << mach->macDA()
335 << ", estimated distance between nodes = " << distance << " m "
336 << endl;
337
338 if (ch->error()) {
339
340 if (debug_)
341 cout << NOW << " UwCsmaAloha_Trigger_NODE(" << addr
342 << ")::Phy2MacEndRx() dropping corrupted pkt " << endl;
343 incrErrorPktsRx();
344
345 refreshReason(UW_CS_ALOHA_TRIG_NODE_REASON_PKT_ERROR);
346 drop(p, 1, UW_CS_ALOHA_TRIG_NODE_DROP_REASON_ERROR);
347 stateRxPacketNotForMe(NULL);
348 } else {
349 if (dest_mac == (int) addr || dest_mac == (int) MAC_BROADCAST) {
350 if (rx_pkt_type == (int) PT_MMAC_TRIGGER) {
351 refreshReason(UW_CS_ALOHA_TRIG_NODE_REASON_TRIGGER_RX);
352 stateRxTrigger(p);
353 } else {
354 refreshReason(UW_CS_ALOHA_TRIG_NODE_REASON_PKT_NOT_FOR_ME);
355 stateRxPacketNotForMe(p);
356 }
357 } else {
358 refreshReason(UW_CS_ALOHA_TRIG_NODE_REASON_PKT_NOT_FOR_ME);
359 stateRxPacketNotForMe(p);
360 }
361 }

364void
365UwCsmaAloha_Trigger_NODE::stateRxTrigger(Packet *p)
366{
367 if (debug_)
368 cout << NOW << "Csma_Aloha_NODE(" << addr
369 << ")::stateRxTrigger---> can_transmit" << can_transmit << endl;
370 if (can_transmit == false) {
371 refreshState(UW_CS_ALOHA_TRIG_NODE_STATE_RX_TRIGGER);
372 can_transmit = true;
373 tx_timer.schedule(tx_timer_duration);
374 if (!Q.empty()) {
375 refreshReason(UW_CS_ALOHA_TRIG_NODE_REASON_LISTEN_AFTER_TRIGGER);
376 stateListen();

379 if (debug_)
380 cout << NOW << "Csma_Aloha_NODE(" << addr
381 << ")::stateRxTrigger---> can_transmit already true " << endl;
382 }
383 Packet::free(p);

431{
432 refreshState(UW_CS_ALOHA_TRIG_NODE_STATE_CHK_LISTEN_TIMEOUT);
433

443 prev_state == UW_CS_ALOHA_TRIG_NODE_STATE_DATA_RX))
444 stateTxData();
445 else
446 stateListen();
447 } else {
448 if (debug_)
449 cerr << NOW << " UwCsmaAloha_Trigger_NODE(" << addr
450 << ")::stateCheckListenExpired() listen_timer logical error, "
451 "current timer state = "
452 << status_info[curr_state] << endl;

489{

492

494

535void
536UwCsmaAloha_Trigger_NODE::stateRxData(Packet *data_pkt)
537{
538 refreshState(UW_CS_ALOHA_TRIG_NODE_STATE_DATA_RX);
539
540 if (debug_)
541 cout << NOW << " UwCsmaAloha_Trigger_NODE(" << addr
542 << ")::stateRxData() " << endl;
543 refreshReason(UW_CS_ALOHA_TRIG_NODE_REASON_DATA_RX);
544
545 switch (prev_state) {
546
547 case UW_CS_ALOHA_TRIG_NODE_STATE_RX_IDLE: {
548 hdr_cmn *ch = hdr_cmn::access(data_pkt);
549 ch->size() = ch->size() - HDR_size;
550 incrDataPktsRx();
551 sendUp(data_pkt);
552 stateIdle();
553 } break;
554
555 case UW_CS_ALOHA_TRIG_NODE_STATE_RX_LISTEN: {
556 hdr_cmn *ch = hdr_cmn::access(data_pkt);
557 ch->size() = ch->size() - HDR_size;
558 incrDataPktsRx();
559 sendUp(data_pkt);
560 stateCheckListenExpired();
561 } break;
562
563 default:
564
565 if (debug_)
566 cerr << NOW << " UwCsmaAloha_Trigger_NODE(" << addr
567 << ")::stateRxData() logical error, prev state = "
568 << status_info[prev_state] << endl;
569 }
