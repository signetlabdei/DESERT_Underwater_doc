225 virtual void
226 freeze()
227 {
228 assert(timer_status == UWSR_RUNNING);
229 left_duration -= (NOW - start_time);
230 if (left_duration <= 0.0)
231 left_duration = module->mac2phy_delay_;
232 force_cancel();
233 timer_status = UWSR_FROZEN;
234 }
235
240 virtual void
241 unFreeze()
242 {
243 assert(timer_status == UWSR_FROZEN);
244 start_time = NOW;
245 assert(left_duration > 0);
246 sched(left_duration);
247 timer_status = UWSR_RUNNING;
248 }
249
253 virtual void
254 stop()
255 {
256 timer_status = UWSR_IDLE;
257 force_cancel();
258 }

264 virtual void
265 schedule(double val)
266 {
267 start_time = NOW;
268 left_duration = val;
269 timer_status = UWSR_RUNNING;
270 resched(val);
271 }
