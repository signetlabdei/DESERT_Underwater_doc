202 virtual void
203 freeze()
204 {
205 assert(timer_status == UWUFETCH_TIMER_STATUS_RUNNING);
206 left_duration = NOW - start_time;
207 if (left_duration <= 0.0) {
208 left_duration = module->mac2phy_delay_;
209 }
210 force_cancel();
211 timer_status = UWUFETCH_TIMER_STATUS_FROZEN;
212 }
213
218 virtual void
219 unFreeze()
220 {
221 assert(timer_status == UWUFETCH_TIMER_STATUS_FROZEN);
222 start_time = NOW;
223 assert(left_duration > 0);
224 sched(left_duration);
228 timer_status = UWUFETCH_TIMER_STATUS_RUNNING;
229 }
230
234 virtual void
235 stop()
236 {
237 timer_status = UWUFETCH_TIMER_STATUS_IDLE;
238 force_cancel();
239 }
240
246 virtual void
247 schedule(double val)
248 {
249 start_time = NOW;
250 left_duration = val;
251 timer_status = UWUFETCH_TIMER_STATUS_RUNNING;
252 resched(val);
253 }
