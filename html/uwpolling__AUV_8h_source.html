307 virtual void
308 freeze()
309 {
310 assert(timer_status == UWPOLLING_RUNNING);
311 left_duration -= (NOW - start_time);
312 if (left_duration <= 0.0) {
313 left_duration = module->mac2phy_delay_;
314 }
315 force_cancel();
316 timer_status = UWPOLLING_FROZEN;
317 }
318
323 virtual void
324 unFreeze()
325 {
326 assert(timer_status == UWPOLLING_FROZEN);
327 start_time = NOW;
328 assert(left_duration > 0);
329 sched(left_duration);
330 timer_status = UWPOLLING_RUNNING;
331 }
332
336 virtual void
337 stop()
338 {
339 timer_status = UWPOLLING_IDLE;
340 force_cancel();
341 }

347 virtual void
348 schedule(double val)
349 {
350 start_time = NOW;
351 left_duration = val;
352 timer_status = UWPOLLING_RUNNING;
353 resched(val);
354 }
